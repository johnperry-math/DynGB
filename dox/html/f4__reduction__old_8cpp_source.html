43 F4_Reduction_Data::F4_Reduction_Data(
44  const list<Critical_Pair_Basic *> & P,
45  const list<Abstract_Polynomial *> & B
46 ) :
47  G(B), Rx(P.front()->first()->base_ring()),
48  M_table(P.front()->first()->base_ring().number_of_variables())
49 {
50  num_cols = 0;
51  A.clear();
52  head.clear();
53  R_build.clear();
54  M_build.clear();
55  strategies.clear();
56  auto mi = M_build.begin();
57  auto ri = R_build.begin();
58  mord = P.front()->first()->monomial_ordering();
59  for (auto p : P) {
60  strategies.push_back(new Poly_Sugar_Data(p->first()));
61  if (strategies.back() != nullptr) {
62  strategies.back()->at_generation_tasks(p->first_multiplier());
63  if (p->second() != nullptr)
64  strategies.back()->pre_reduction_tasks(p->second_multiplier().log(), *(p->second()));
65  }
66  add_monomials(mi, ri, p->first(), p->first_multiplier(), true);
67  if (p->second() != nullptr) {
68  *ri = const_cast<Abstract_Polynomial *>(p->second());
69  ++ri; ++mi;
70  add_monomials(mi, ri, p->second(), p->second_multiplier());
71  }
72  }
73  // for each monomial, find an appropriate reducer
74  mi = M_build.begin();
75  ri = R_build.begin();
76  while (mi != M_build.end()) {
77  auto g = G.begin();
78  bool found = ((*ri) != nullptr);
79  while (not found and g != G.end()) {
80  if ((**mi).divisible_by((*g)->leading_monomial())) {
81  found = true;
82  *ri = *g;
83  Monomial u(**mi);
84  u /= (*g)->leading_monomial();
85  add_monomials(mi, ri, *g, u);
86  g = G.end();
87  }
88  ++g;
89  }
90  ++mi;
91  ++ri;
92  }
93  initialize_many(P);
94  unsigned els = 0;
95  for (auto Ak : A) els += Ak.size();
96  cout << "saved " << (num_rows*num_cols - els)*100 / (num_rows*num_cols) << "%\n";
97 }

99 void F4_Reduction_Data::initialize_some_rows(
100  const list<Critical_Pair_Basic *> & P, unsigned row
101 ) {
102  const unsigned num_cols = M_build.size();
103  const COEF_TYPE F0 = 0;
104  for (auto cp : P) {
105  auto p = cp->first();
106  const Monomial & t = cp->first_multiplier();
107  auto pi = p->new_iterator();
108  vector<COEF_TYPE> & Arow = A[row];
109  nonzero_entries[row] = 0;
110  unsigned i = M_table.lookup_product(pi->currMonomial(), t);
111  Arow.resize(num_cols - i, F0);
112  Arow[0] = pi->currCoeff().value();
113  offset[row] = i;
114  head[row] = 0;
115  nonzero_entries[row] = 1;
116  pi->moveRight();
117  for (/* */; not pi->fellOff(); ++i) {
118  i = M_table.lookup_product(pi->currMonomial(), t);
119  Arow[i - offset[row]] = pi->currCoeff().value();
120  pi->moveRight();
121  ++nonzero_entries[row];
122  }
123  delete pi;
124  strategies[row] = new Poly_Sugar_Data(cp->first());
125  delete cp;
126  ++row;
127  }

185 void F4_Reduction_Data::add_monomials(
186  list<Monomial *>::iterator & t1,
187  list<Abstract_Polynomial *>::iterator & r1,
188  const Abstract_Polynomial *g,
189  const Monomial & u,
190  bool new_row
191 ) {
192  NVAR_TYPE n = g->leading_monomial().num_vars();
193  if (new_row) {
194  t1 = M_build.begin();
195  r1 = R_build.begin();
196  Monomial * t = new Monomial(g->leading_monomial());
197  (*t) *= u;
198  find_position(t1, M_build.end(), r1, *t);
199  if (t1 == M_build.end()) {
200  M_build.push_back(t);
201  R_build.push_back(nullptr);
202  t1 = M_build.begin();
203  r1 = R_build.begin();
204  auto t2(t1);
205  ++t2;
206  while (t2 != M_build.end()) {
207  ++t1; ++r1;
208  ++t2;
209  }
210  } else if (**t1 != *t) {
211  M_build.insert(t1, t);
212  R_build.insert(r1, nullptr);
213  --t1; --r1;
214  }
215  }
216  auto t2(t1); auto r2(r1);
217  Polynomial_Iterator * pi = g->new_iterator();
218  pi->moveRight();
219  while (not (pi->fellOff())) {
220  Monomial * t = new Monomial(pi->currMonomial());
221  (*t) *= u;
222  find_position(t2, M_build.end(), r2, *t);
223  if (t2 == M_build.end()) {
224  M_build.push_back(t);
225  R_build.push_back(nullptr);
226  } else if (**t2 != *t) {
227  M_build.insert(t2, t);
228  R_build.insert(r2, nullptr);
229  }
230  pi->moveRight();
231  }
232  delete pi;
233 }

281 void F4_Reduction_Data::reduce_my_rows(
282  const set<unsigned> & my_rows
283 ) {
284  NVAR_TYPE n = Rx.number_of_variables();
285  const Prime_Field & F = Rx.ground_field();
286  red_mutex.lock();
287  Monomial u(n);
288  red_mutex.unlock();
289  UCOEF_TYPE mod = F.modulus();
290  unsigned new_nonzero_entries;
291  for (unsigned mi = 0; mi < num_cols; ++mi) {
292  // is this monomial reducible?
293  if (R[mi] != nullptr) {
294  for (unsigned k : my_rows) {
295  vector<COEF_TYPE> & Ak = A[k];
296  // do we need to reduce this poly?
297  if (offset[k] <= mi and Ak[mi - offset[k]] != 0) {
298  unsigned i = mi - offset[k];
299  // get reducer for this monomial
300  const Abstract_Polynomial * g = R[mi];
301  Polynomial_Iterator * gi = g->new_iterator();
302  // determine multiplier
303  Monomial & t = *M[mi];
304  for (NVAR_TYPE l = 0; l < n; ++l)
305  u.set_exponent(l, t[l] - (gi->currMonomial())[l]);
306  red_mutex.lock();
307  vector<pair<unsigned, COEF_TYPE> > & r = R_built[mi];
308  if (r.size() == 0) { // need to create reducer
309  size_t j = mi;
310  // loop through g's terms
311  while (not gi->fellOff()) {
312  const Monomial & t = gi->currMonomial();
313  j = M_table.lookup_product(u, t);
314  r.emplace_back(j, gi->currCoeff().value());
315  gi->moveRight();
316  }
317  }
318  delete gi;
319  red_mutex.unlock();
320  // prepare for reduction
321  new_nonzero_entries = nonzero_entries[k];
322  auto sk = strategies[k];
323  sk->pre_reduction_tasks(u, *g);
324  // determine reduction coefficient
325  COEF_TYPE a(mod - ((Ak[i]*F.inverse(r[0].second)) % mod));
326  // loop through g's terms
327  for (auto ri : r) {
328  unsigned j = ri.first;
329  unsigned l = j - offset[k];
330  bool was_zero = Ak[l] == 0;
331  Ak[l] += a*ri.second; Ak[l] %= mod;
332  if (was_zero)
333  ++new_nonzero_entries;
334  else if (Ak[l] == 0)
335  --new_nonzero_entries;
336  // advance
337  }
338  unsigned & hk = head[k];
339  while (hk < Ak.size() and Ak[hk] == 0) ++hk;
340  nonzero_entries[k] = new_nonzero_entries;
341  }
342  }
343  }
344  }

367 void F4_Reduction_Data::reduce_my_new_rows(
368  unsigned i,
369  const Prime_Field & F,
370  const set<unsigned> & to_reduce
371 ) {
372  auto mod = F.modulus();
373  const auto & Ai = A[i];
374  for (auto j : to_reduce) {
375  auto ci = head[i];
376  auto & Aj = A[j];
377  auto c = ci + offset[i] - offset[j]; // pos in A[j] of A[i]'s head
378  auto a = Aj[c];
379  unsigned ops = 0;
380  a *= mod - F.inverse(Ai[head[i]]);
381  while (c < Aj.size() and ops < nonzero_entries[i]) {
382  if (Ai[ci] != 0) {
383  bool was_zero = (Aj[c] == 0);
384  Aj[c] += a*Ai[ci]; Aj[c] %= mod;
385  if (was_zero)
386  ++nonzero_entries[j];
387  else if (Aj[c] == 0)
388  --nonzero_entries[j];
389  ++ops;
390  }
391  ++c; ++ci;
392  }
393  unsigned & hj = head[j];
394  while (hj < Aj.size() and Aj[hj] == 0) ++hj;
395  }
396 }

398 void F4_Reduction_Data::reduce_by_new() {
399  Prime_Field F(Rx.ground_field());
400  unsigned cores = std::thread::hardware_concurrency() * 2;
401  unsigned num_threads = (cores < num_rows) ? cores : num_rows;
402  set<unsigned> * thread_rows = new set<unsigned>[num_threads];
403  thread * workers = new thread[num_threads];
404  for (unsigned i = 0; i < num_rows; ++i) {
405  if (nonzero_entries[i] > 0) {
406  COEF_TYPE Ai0 = head[i] + offset[i]; // abs pos of A[i]'s head
407  for (unsigned j = 0; j < num_threads; ++j)
408  thread_rows[j].clear();
409  unsigned k = 0;
410  for (unsigned j = 0; j < num_rows; ++j) {
411  if (
412  j != i and Ai0 > offset[j]
413  and (A[j][Ai0 - offset[j]] != 0)
414  ) {
415  thread_rows[k].insert(j);
416  ++k; k %= num_threads;
417  }
418  }

420  workers[c] = thread(
421  &F4_Reduction_Data::reduce_my_new_rows, this,

432 vector<Constant_Polynomial *> F4_Reduction_Data::finalize() {
433  cout << "spent " << copy_time << " seconds copying\n";
434  vector<Constant_Polynomial *> result;
435  const Prime_Field & F = Rx.ground_field();
436  UCOEF_TYPE mod = F.modulus();
437  NVAR_TYPE n = M[0]->num_vars();
438  for (unsigned i = 0; i < num_rows; ++i) {
439  if (nonzero_entries[i] == 0) {
440  delete strategies[i];
441  strategies[i] = nullptr;
442  } else {
443  vector<COEF_TYPE> & Ai = A[i];
444  Monomial * M_final = static_cast<Monomial *>(
445  malloc(sizeof(Monomial)*nonzero_entries[i])
446  );
447  Prime_Field_Element * A_final
448  = static_cast<Prime_Field_Element *>(
449  malloc(sizeof(Prime_Field_Element)*nonzero_entries[i])
450  );
451  COEF_TYPE scale = F.inverse(Ai[head[i]]);
452  unsigned k = 0;
453  for (unsigned j = head[i]; k < nonzero_entries[i]; ++j) {
454  if (Ai[j] != 0) {
455  A_final[k].assign(Ai[j]*scale % mod, &F);
456  M_final[k].common_initialization();
457  M_final[k].initialize_exponents(n);
458  M_final[k].set_monomial_ordering(mord);
459  M_final[k] = *M[offset[i] + j];
460  ++k;
461  }
462  }
463  result.push_back(new Constant_Polynomial(
464  nonzero_entries[i],
465  Rx,
466  M_final, A_final,
467  mord
468  ));
469  result.back()->set_strategy(strategies[i]);
470  strategies[i] = nullptr;
471  free(M_final);
472  free(A_final);
473  }
474  }
475  return result;
476 }

478 list<Constant_Polynomial *> f4_control(const list<Abstract_Polynomial *> &F) {
479  time_t start_f4 = time(nullptr);
480  cout << "computation started at " << asctime(localtime(&start_f4)) << endl;
481  unsigned number_of_spolys = 0;
482  double total_time = 0;
483  list<Abstract_Polynomial *> G;
484  list<Critical_Pair_Basic *> P;
485  // set up basis with generators
486  for (Abstract_Polynomial * fo : F)
487  {
488  Constant_Polynomial * f = new Constant_Polynomial(*fo);
489  f->set_strategy(new Poly_Sugar_Data(f));
490  f->strategy()->at_generation_tasks();
491  P.push_back(new Critical_Pair_Basic(f, StrategyFlags::SUGAR_STRATEGY));
492  }
493  // main loop
494  bool verbose = false;
495  bool very_verbose = false;
496  list<Critical_Pair_Basic *> Pnew;
497  while (not P.empty()) {
498  sort_pairs_by_strategy(P);
499  report_critical_pairs(P);
500  Critical_Pair_Basic * p = P.front();
501  report_front_pair(p, StrategyFlags::SUGAR_STRATEGY);
502  cout << "\tdegree: " << p->lcm().total_degree(0) << endl;
503  P.pop_front();
504  Pnew.push_back(p);
505  DEG_TYPE mindeg = p->lcm().total_degree();
506  for (auto pi = P.begin(); pi != P.end(); /* */) {
507  p = *pi;
508  if (p->lcm().total_degree() < mindeg) {
509  for (auto qi = Pnew.begin(); qi != Pnew.end(); ++qi)
510  P.push_front(*qi);
511  Pnew.clear();
512  mindeg = p->lcm().total_degree();
513  report_front_pair(p, StrategyFlags::SUGAR_STRATEGY);
514  cout << "\tdegree: " << p->lcm().total_degree(0) << endl;
515  Pnew.push_back(p);
516  auto qi = pi;
517  ++qi;
518  P.erase(pi);
519  pi = qi;
520  } else if (p->lcm().total_degree() == mindeg) {
521  report_front_pair(p, StrategyFlags::SUGAR_STRATEGY);
522  Pnew.push_back(p);
523  auto qi = pi;
524  ++qi;
525  P.erase(pi);
526  pi = qi;
527  } else
528  ++pi;
529  }
530  // make s-poly
531  //time_t start_time = time(nullptr);
532  F4_Reduction_Data s(Pnew, G); // cyc8h ~20sec
533  //time_t end_time = time(nullptr);

535  number_of_spolys += Pnew.size();
536  Pnew.clear();
537  if (not s.is_zero()) {
538  s.reduce_by_old(); // cyc8h ~38sec
539  s.reduce_by_new(); // cyc8h ~4sec
540  }
541  if (s.is_zero()) {
542  cout << "\tmatrix reduced to zero\n";

552  if (very_verbose) { cout << "\tadded "; r->println(); }

558  // cleanup

562  cout << G.size() << " polynomials after interreduction\n";
563  list<Constant_Polynomial *> B;
564  for (Abstract_Polynomial * g : G)
565  B.push_back(new Constant_Polynomial(*g));
566  time_t end_f4 = time(nullptr);
567  double duration = difftime(end_f4, start_f4);
568  cout << "computation ended at " << asctime(localtime(&end_f4)) << endl;
