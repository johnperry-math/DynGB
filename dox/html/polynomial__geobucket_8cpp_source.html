103 void Geobucket_Iterator::moveRight() {
104  if (pi != nullptr) {
105  pi->moveRight();
106  if (pi->fellOff()) {
107  delete pi;
108  pi = nullptr;
109  do
110  ++bucket_number;
111  while (bucket_number < NUM_BUCKETS
112  and g->buckets[bucket_number] == nullptr);
113  if (bucket_number < NUM_BUCKETS and g->buckets[bucket_number] != nullptr)
114  pi = new LLPolynomial_Iterator(g->buckets[bucket_number]);
115  }
116  }
117 }

213 void Polynomial_Geobucket::recompute_leading_monomial() {
214  unsigned i = 0;
215  do {
216  i = 0;
217  for (unsigned j = 1; j < NUM_BUCKETS; ++j)
218  if (i != j and buckets[j] != nullptr and (not buckets[j]->is_zero())) {
219  if (buckets[i]->is_zero() or
220  buckets[j]->leading_monomial() > buckets[i]->leading_monomial())
221  i = j;
222  else {
223  if (buckets[i]->leading_monomial().is_like(
224  buckets[j]->leading_monomial())) {
225  Polynomial_Linked_List * q = buckets[j]->detach_head();
226  *buckets[i] += *q;
227  delete q;
228  j = 0; // need to restart from bucket 1
229  if (buckets[i]->is_zero()) {
230  delete buckets[i];
231  buckets[i] = nullptr;
232  i = 0;
233  }
234  }
235  }
236  }
237  } while (i != 0 and buckets[i]->is_zero());
238  if (i == 0)
239  delete buckets[0]->detach_head();
240  else {
241  if (buckets[0] != nullptr)
242  delete buckets[0];
243  buckets[0] = buckets[i]->detach_head();
244  }
245  // condensation does not seem to help
246  /*++aops_performed;
247  if (aops_performed > 50) {
248  aops_performed = 0;
249  condense_buckets();
250  }*/
251 }

270 Polynomial_Geobucket & Polynomial_Geobucket::operator +=(
271  const Abstract_Polynomial & g
272 ) {
273  Polynomial_Linked_List * f = buckets[0];
274  buckets[0] = new Polynomial_Linked_List(base_ring());
275  (*f) += g;
276  unsigned i = lglen(f->length());
277  unsigned m = i;
278  bool placed = false;
279  while (not placed and i < NUM_BUCKETS) {
280  if (buckets[i] != nullptr and not buckets[i]->is_zero()) {
281  (*f) += *(buckets[i]);
282  m = lglen(f->length());
283  }
284  if (m < (BUCKET_BASE << BUCKET_SHIFT * i)) {
285  placed = true;
286  if (not f->is_zero()) {
287  if (buckets[i] != nullptr)
288  delete buckets[i];
289  buckets[i] = f;
290  }
291  }
292  }
293  recompute_leading_monomial();
294  return *this;
295 }

322 void Polynomial_Geobucket::add_polynomial_multiple(
323  const Prime_Field_Element & b, const Monomial & u,
324  const Abstract_Polynomial & g, bool subtract
325  )
326 {
327  unsigned i = lglen(g.length() + 1);
328  if (i >= NUM_BUCKETS) throw(new runtime_error("too large for buckets"));
329  Polynomial_Linked_List * f = buckets[0];
330  f->add_polynomial_multiple(b, u, g, subtract);
331  buckets[0] = new Polynomial_Linked_List(base_ring());
332  if (buckets[i] == nullptr)
333  buckets[i] = f;
334  else if (buckets[i]->is_zero()) {
335  delete buckets[i];
336  buckets[i] = f;
337  } else {
338  (*f) += *(buckets[i]);
339  delete buckets[i];
340  buckets[i] = nullptr;
341  while (i < NUM_BUCKETS and lglen(f->length()) > i)
342  {
343  ++i;
344  if (buckets[i] != nullptr) {
345  (*f) += *(buckets[i]);
346  delete buckets[i];
347  buckets[i] = new Polynomial_Linked_List(base_ring());
348  }
349  }
350  if (i < NUM_BUCKETS) {
351  if (buckets[i] != nullptr)
352  delete buckets[i];
353  buckets[i] = f;
354  }
355  else
356  throw(new runtime_error("too large for buckets"));
357  }
358  recompute_leading_monomial();
359 }

368 void Polynomial_Geobucket::add_last(
369  const Prime_Field_Element & a, const Monomial & t
370 ) {

404 void Polynomial_Geobucket::condense_buckets() {
405  if (not is_zero()) {
406  Polynomial_Linked_List * f;
407  unsigned i = 1;
408  while (i < NUM_BUCKETS and (buckets[i] == nullptr or buckets[i]->is_zero()))
409  ++i;
410  if (i < NUM_BUCKETS) {
411  f = new Polynomial_Linked_List(*(buckets[i]));
412  buckets[i] = new Polynomial_Linked_List(base_ring());
413  ++i;
414  while (i < NUM_BUCKETS) {
415  if (buckets[i] != nullptr and (not buckets[i]->is_zero())) {
416  *f += *(buckets[i]);
417  delete buckets[i];
418  buckets[i] = new Polynomial_Linked_List(base_ring());
419  }
420  ++i;
421  }
422  unsigned i = lglen(f->length() + 1);
423  if (i >= NUM_BUCKETS) throw(new runtime_error("too large for buckets"));
424  delete buckets[i];
425  buckets[i] = f;
426  }
