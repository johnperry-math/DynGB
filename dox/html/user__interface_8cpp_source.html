58 void user_interface() {
59  UCOEF_TYPE p;
60  cout << "prime number for base field: ";
61  cin >> p;
62  NVAR_TYPE n;
63  cout << "number of indeterminates: ";
64  cin >> n;
65  char specify_names;
66  cout << "do you want to specify the indeterminates' names? (y/n) ";
67  cin >> specify_names;
68  string * names = new string[n];
69  Prime_Field F(p);
70  Polynomial_Ring * P;
71  if (not specify_names) {
72  P = new Polynomial_Ring(n, F);
73  cout << "indeterminate names are: ";
74  for (NVAR_TYPE i = 0; i < n; ++i) {
75  cout << P->name(i);
76  names[i] = P->name(i);
77  }
78  } else {
79  for (NVAR_TYPE i = 0; i < n; ++i)
80  cin >> names[i];
81  P = new Polynomial_Ring(n, F, names);
82  }
83  unsigned m;
84  cout << "number of polynomials: ";
85  cin >> m;
86  cin.ignore();
87  list<Abstract_Polynomial *> I;
88  for (unsigned np = 0; np < m; ++np) {
89  cout << "please enter polynomial #" << np << ": (use spaces to separate factors) ";
90  string inpoly;
91  getline(cin, inpoly);
92  list<Monomial> M; // monomials
93  list<Prime_Field_Element> A; // coefficients
94  unsigned nt = 0; // number of terms
95  unsigned i = 0;
96  while (i < inpoly.size()) {
97  bool reading_term = true;
98  COEF_TYPE a = 1;
99  EXP_TYPE exp[n];
100  for (NVAR_TYPE i = 0; i < n; ++i) exp[i] = 0;
101  while (inpoly[i] == ' ') ++i;
102  while (reading_term and i < inpoly.size()) {
103  unsigned j = i;
104  if (
105  inpoly[i] == '+' or inpoly[i] == '-' or
106  (inpoly[i] >= '0' and inpoly[i] <= '9')
107  ) {
108  bool positive = true;
109  if (inpoly[i] == '-') positive = false;
110  if (inpoly[i] == '+' or inpoly[i] == '-') ++i;
111  while (inpoly[i] == ' ') ++i;
112  if (inpoly[i] >= '0' and inpoly[i] <= '9') {
113  j = i;
114  while (inpoly[j] >= '0' and inpoly[j] <= '9') ++j;
115  a *= stol(inpoly.substr(i, j - i));
116  if (not positive) a *= -1;
117  i = j;
118  }
119  }
120  else if (inpoly[i] == '*')
121  ++i;
122  else if (inpoly[i] == '\0')
123  ++i;
124  else { // should be characters
125  unsigned j = i;
126  while ((inpoly[j] >= 'a' and inpoly[j] <= 'z') or
127  (inpoly[j] >= 'A' and inpoly[j] <= 'Z') or
128  (inpoly[j] >= '0' and inpoly[j] <= '9'))
129  ++j;
130  string var_name = inpoly.substr(i, j - i);
131  bool found = false;
132  unsigned k = 0;
133  for (/* */; not found and k < n; ++k)
134  if (var_name.compare(names[k]) == 0) {
135  found = true;
136  --k;
137  }
138  i = j;
139  while (inpoly[i] == ' ') ++i;
140  if (inpoly[i] != '^')
141  exp[k] += 1;
142  else {
143  ++i;
144  while (inpoly[i] == ' ') ++i;
145  unsigned j = i;
146  while (inpoly[j] >= '0' and inpoly[j] <= '9') ++j;
147  exp[k] += stol(inpoly.substr(i, j - i));
148  i = j;
149  }
150  }
151  while (inpoly[i] == ' ') ++i;
152  if (inpoly[i] == '+' or inpoly[i] == '-') reading_term = false;
153  }
154  ++nt;
155  Monomial t(n, exp);
156  M.push_back(t);
157  A.emplace_back(a, &F);
158  }
159  Abstract_Polynomial * p = new Constant_Polynomial(*P, M, A);
160  p->sort_by_order();
161  cout << "read " << *p << endl;
162  I.push_back(p);
163  }
164  string computation;
165  while (
166  computation.compare("s") and computation.compare("d")
167  and computation.compare("static") and computation.compare("dynamic")
168  ) {
169  cout << "static (s) or dynamic (d) computation? ";
170  getline(cin, computation);
171  }
172  list<Constant_Polynomial *> B;
173  bool dynamic = not (computation.compare("d") and computation.compare("dynamic"));
174  if (not dynamic) {
175  B = buchberger(I);
176  } else {
177  DynamicSolver solver;
178  string solver_choice;
179  while (
180  solver_choice.compare("skel") and solver_choice.compare("glpk")
181  and solver_choice.compare("ppl") and solver_choice.compare("skeleton")
182  ) {
183  cout << "which solver? ([skel]eton, glpk, ppl) ";
184  getline(cin, solver_choice);
185  }
186  if (not solver_choice.compare("skel") or not solver_choice.compare("skeleton"))
187  solver = SKELETON_SOLVER;
188  else if (not solver_choice.compare("glpk"))
189  solver = GLPK_SOLVER;
190  else if (not solver_choice.compare("ppl"))
191  solver = PPL_SOLVER;
192  string heur_choice;
193  Dynamic_Heuristic heuristic;
194  while (
195  heur_choice.compare("h") and heur_choice.compare("gh")
196  and heur_choice.compare("b") and heur_choice.compare("bb")
197  and heur_choice.compare("gb") and heur_choice.compare("c")
198  and heur_choice.compare("d") and heur_choice.compare("r")
199  ) {
200  cout << "available heuristics are\n";
201  cout << "\t h = hilbert with standard grading\n";
202  cout << "\tgh = hilbert with order-based grading\n";
203  cout << "\t b = betti with standard grading\n";
204  cout << "\tbb = \"big\" betti with standard grading\n";
205  cout << "\tgb = betti with order-based grading\n";
206  cout << "\t c = minimal number of critical pairs\n";
207  cout << "\t d = minimal degree, ties broken by hilbert\n";
208  cout << "\t r = random choice\n";
209  cout << "which heuristic would you like? ";
210  getline(cin, heur_choice);
211  }
212  if (not heur_choice.compare( "h")) heuristic = Dynamic_Heuristic::ORD_HILBERT_THEN_DEG;
213  else if (not heur_choice.compare("gh")) heuristic = Dynamic_Heuristic::GRAD_HILB_THEN_DEG;
214  else if (not heur_choice.compare( "b")) heuristic = Dynamic_Heuristic::BETTI_HILBERT_DEG;
215  else if (not heur_choice.compare("bb")) heuristic = Dynamic_Heuristic::BIG_BETTI_HILBERT_DEG;
216  else if (not heur_choice.compare("gb")) heuristic = Dynamic_Heuristic::GRAD_BETTI_HILBERT_DEG;
217  else if (not heur_choice.compare( "c")) heuristic = Dynamic_Heuristic::MIN_CRIT_PAIRS;
218  else if (not heur_choice.compare( "d")) heuristic = Dynamic_Heuristic::DEG_THEN_ORD_HILBERT;
219  else if (not heur_choice.compare( "r")) heuristic = Dynamic_Heuristic::EVIL_RANDOM;
220  string whether_analysis;
221  bool analyze_first = false;
222  while (whether_analysis.compare("y") and whether_analysis.compare("n")) {
223  cout << "perform global analysis at the outset? (y or n) ";
224  cin >> whether_analysis;
225  }
226  if (not whether_analysis.compare("y")) analyze_first = true;
227  B = buchberger_dynamic(
228  I, SPolyCreationFlags::GEOBUCKETS, StrategyFlags::SUGAR_STRATEGY,
229  nullptr, heuristic, solver, analyze_first
230  );
231  }
232  //check_correctness(B, StrategyFlags::NORMAL_STRATEGY);
233  cout << "have basis with " << B.size() << " elements:\n";
234  for (auto b : B) {
235  cout << b->leading_monomial() << ", ";
236  delete b;
237  }
238  cout << endl;
239  for (auto p : I) delete p;
240  delete P;
241  delete [] names;
242 }
