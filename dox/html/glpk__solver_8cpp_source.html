29 GLPK_Solver::GLPK_Solver(NVAR_TYPE nx) {
30  dirty = true;
31  n = nx; m = 0;
32  lp = glp_create_prob();
33  glp_set_obj_dir(lp, GLP_MIN);
34  glp_add_cols(lp, n);
35  for (NVAR_TYPE i = 1; i <= n; ++i) {
36  glp_set_col_bnds(lp, i, GLP_LO, MIN_X, 0.0);
37  glp_set_obj_coef(lp, i, 1.0);
38  }
39  glp_init_smcp(&smcp);
40  glp_term_out(GLP_OFF);
41  smcp.msg_lev = GLP_MSG_OFF;
42  smcp.presolve = GLP_ON;
43  row_data = new double[1+n]; // GLPK wants to start from index 1
44  row_indx = new int[1+n];
45  ray_data = new RAYENT_TYPE[n];
46 }

96 bool GLPK_Solver::solve(const vector<Constraint> & newvecs) {
97  dirty = true;
98  int glp_result = 0;
99  int new_m = newvecs.size();
100  if (new_m > 0) {
101  glp_add_rows(lp, new_m);
102  for (int i = 0; i < new_m; ++i) {
103  int num_valid = 0;
104  for (int k = 0; k < n; ++k) {
105  if (newvecs[i][k] != 0) {
106  ++num_valid;
107  row_indx[num_valid] = 1 + k;
108  row_data[num_valid] = newvecs[i][k];
109  }
110  }
111  glp_set_mat_row(lp, 1 + m + i, num_valid, row_indx, row_data);
112  glp_set_row_bnds(lp, 1 + m + i, GLP_LO, MIN_X, 0.0);
113  }
114  if (m != 0) {
115  smcp.presolve = GLP_OFF;
116  glp_warm_up(lp);
117  }
118  m += new_m;
119  glp_result = glp_simplex(lp, &smcp);
120  if (glp_result == 0)
121  glp_result = glp_exact(lp, nullptr);
122  }
123  int status = glp_get_status(lp);
124  return (glp_result == 0 and (status == GLP_OPT or status == GLP_FEAS));
125 }

127 bool GLPK_Solver::solve(const Constraint & newvec) {
128  dirty = true;
129  glp_add_rows(lp, 1);
130  int num_valid = 0;
131  for (int k = 0; k < n; ++k) {
132  if (newvec[k] != 0) {
133  ++num_valid;
134  row_indx[num_valid] = 1 + k;
135  row_data[num_valid] = newvec[k];
136  }
137  }
138  glp_set_mat_row(lp, 1 + m, num_valid, row_indx, row_data);
139  glp_set_row_bnds(lp, 1 + m, GLP_LO, MIN_X, 0.0);
140  if (m != 0) {
141  smcp.presolve = GLP_OFF;
142  glp_warm_up(lp);
143  }
144  m += 1;
145  int glp_result = glp_simplex(lp, &smcp);
146  smcp.presolve = GLP_OFF;

153 const set<Ray> & GLPK_Solver::get_rays() const {
154  if (dirty) {
155  rays.clear();
156  // the next few lines add a row
157  // that pushes the solution beyond the actual minimum
158  double curr_min = glp_get_obj_val(lp);
159  glp_add_rows(lp, 1);
160  for (NVAR_TYPE i = 0; i < n; ++i) {
161  row_indx[1+i] = 1+i;
162  row_data[1+i] = 1.0;
163  }
164  glp_set_mat_row(lp, 1 + m, n, row_indx, row_data);
165  glp_set_row_bnds(lp, 1 + m, GLP_DB, curr_min + 1, curr_min + 2);
166  // now we vary the objective function to minimize each variable
167  for (int i = 0; i < n; ++i)
168  glp_set_obj_coef(lp, 1 + i, 0.0);
169  for (int i = 0; i < n; ++i) {
170  glp_set_obj_coef(lp, 1 + i, 1.0);
171  if (i > 0)
172  glp_set_obj_coef(lp, i, 0.0);
173  // minimize first
174  glp_simplex(lp, &smcp);
175  glp_get_status(lp);
176  glp_exact(lp, nullptr);
177  for (NVAR_TYPE j = 0; j < n; ++j)
178  ray_data[j] = static_cast<RAYENT_TYPE>(
179  round(1/MIN_X * glp_get_col_prim(lp, 1 + j))
180  );
181  rays.emplace(n, ray_data);
182  //now maximize
183  glp_set_obj_dir(lp, GLP_MAX);
184  glp_simplex(lp, &smcp);
185  glp_get_status(lp);
186  glp_exact(lp, nullptr);
187  for (NVAR_TYPE j = 0; j < n; ++j)
188  ray_data[j] = static_cast<RAYENT_TYPE>(
189  round(1/MIN_X * glp_get_col_prim(lp, 1 + j))
190  );
191  rays.emplace(n, ray_data);
192  glp_set_obj_dir(lp, GLP_MIN);
193  }
194  // fix the objective function
195  for (unsigned i = 0; i < n; ++i)
196  glp_set_obj_coef(lp, 1 + i, 1.0);
197  // delete the row we added
198  row_indx[1] = 1 + m;
199  glp_set_row_stat(lp, 1 + m, GLP_BS);
200  glp_del_rows(lp, 1, row_indx);
201  glp_std_basis(lp);
202  dirty = false;
203  }
204  return rays;
205 }
