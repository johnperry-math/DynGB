240  if (grading != current_grading)
241  delete hGradNum;
242  hGradNum = hilbert_numerator_bigatti(gens, grading);
243  current_grading = grading;

341  void add_generator(const Monomial & t) {
342  if (hNum != nullptr) {
343  Monomial_Ideal J(t.num_vars(), colon_ideal_without_ideals(gens, t));
344  Dense_Univariate_Integer_Polynomial * hn_new = J.hilbert_numerator();
345  J.forget_hilbert_numerator();
346  hn_new->multiply_by_monomial_of_degree(t.total_degree());
347  hn_new->negate();
348  hn_new->add(*hNum);
349  delete hNum;
350  hNum = hn_new;
351  if (hRedNum != nullptr) delete hRedNum;
352  if (hPol != nullptr) delete hPol;
353  }
354  gens.push_back(t);
355  }
