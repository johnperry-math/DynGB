67 bool Generic_Grevlex::first_larger(
68  const Monomial & t, const Monomial & u
69 ) const {
70  DEG_TYPE dtk = t.ordering_degree();
71  DEG_TYPE duk = u.ordering_degree();
72  bool searching = dtk == duk;
73  if (searching) {
74  NVAR_TYPE n = t.num_vars();
75  const EXP_TYPE * a = t.log();
76  const EXP_TYPE * b = u.log();
77  NVAR_TYPE k = 0;
78  for (/* */; searching and k < n; ++k) {
79  dtk -= a[n - k - 1];
80  duk -= b[n - k - 1];
81  searching = dtk == duk;
82  }
83  }
84  return dtk > duk;
85 }

437  int result;

566 bool nonsingular(NVAR_TYPE m, NVAR_TYPE n, const WT_TYPE **A) {
567  bool possibly = true;
568  // first copy A
569  long long ** M = new long long * [m];
570  for (NVAR_TYPE i = 0; i < m; ++i) {
571  M[i] = new long long [n];
572  for (NVAR_TYPE j = 0; j < n; ++j)
573  M[i][j] = A[i][j];
574  }
575  std::cout << m << ',' << n << std::endl;
576  for (NVAR_TYPE i = 0; i < m; ++i) {
577  for (NVAR_TYPE j = 0; j < n; ++j)
578  std::cout << M[i][j] << ' ';
579  std::cout << std::endl;
580  }
581  std::cout << std::endl;
582  for (NVAR_TYPE i = 0; possibly and i < m; ++i) {
583  // first find a nonzero element in this column
584  bool searching = true;
585  NVAR_TYPE j = i;
586  for (/* */; searching and j < m; ++j)
587  if (M[j][i] != 0) {
588  searching = false;
589  --j;
590  }
591  possibly = not searching;
592  if (possibly and j < m) {
593  // check if we need to swap rows
594  if (j != i) {
595  for (NVAR_TYPE k = i; k < n; ++k) {
596  long long temp = M[j][k];
597  M[j][k] = M[i][k];
598  M[i][k] = temp;
599  }
600  }
601  // clear out the rest of the column
602  for (j = i + 1; j < m; ++j) {
603  WT_TYPE header = M[j][i];
604  if (header != 0) {
605  // delete header & adjust row
606  for (NVAR_TYPE k = i; k < n; ++k)
607  M[j][k] = M[j][k] * M[i][i] - M[i][k] * header;
608  }
609  }
610  }
611  }
612  std::cout << m << ',' << n << std::endl;
613  for (NVAR_TYPE i = 0; i < m; ++i) {
614  for (NVAR_TYPE j = 0; j < n; ++j)
615  std::cout << M[i][j] << ' ';
616  std::cout << std::endl;
617  }
618  std::cout << std::endl;
619  // free memory
620  for (NVAR_TYPE i = 0; i < m; ++i)
621  delete [] M[i];
622  delete [] M;
623  // should be no need to check main diagonal explicitly;
624  // obtaining zero in a main diagonal should mean possibly is now false
625  return possibly;
626 }
