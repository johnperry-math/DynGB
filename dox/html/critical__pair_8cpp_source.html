106 Mutable_Polynomial * Critical_Pair_Basic::s_polynomial(
107  SPolyCreationFlags method, StrategyFlags strategy
108 ) {
109  if (s != nullptr) return s;
110  if (key != nullptr) key->pre_spolynomial_tasks();
111  // create multiple of first() according to the indicated method
112  switch (method) {
113  case SPolyCreationFlags::LINKED_LST:
114  s = new Polynomial_Linked_List(*p);
115  break;
116  case SPolyCreationFlags::GEOBUCKETS:
117  s = new Polynomial_Geobucket(*p);
118  break;
119  case SPolyCreationFlags::DOUBLE_BUF:
120  s = new Double_Buffered_Polynomial(*p);
121  break;
122  default:
123  s = new Polynomial_Geobucket(*p);
124  break;
125  }
126  //cout << "s-poly: " << *s << endl;
127  switch(strategy) {
128  case StrategyFlags::NORMAL_STRATEGY:
129  break; /* nothing to do */
130  case StrategyFlags::SUGAR_STRATEGY :
131  s->set_strategy(new Poly_Sugar_Data(s)); break;
132  case StrategyFlags::WSUGAR_STRATEGY: {
133  Poly_WSugar_Data * sd = static_cast<Poly_WSugar_Data *>(p->strategy());
134  const WT_TYPE * w;
135  if (sd == nullptr) w = nullptr;
136  else
137  w = (static_cast<const Weighted_Ordering *>(p->monomial_ordering()))
138  ->order_weights();
139  s->set_strategy(new Poly_WSugar_Data(s, w));
140  break;
141  }
142  default: break; /* default to normal strategy */
143  }
144  if (s->strategy() != nullptr) { s->strategy()->at_generation_tasks(tp); }
145  s->multiply_by_monomial(tp);
146  //cout << "after multiply by monomial " << tp << " : " << *s << endl;
147  if (q != nullptr) {
148  Prime_Field_Element a = s->leading_coefficient();
149  if (not q->leading_coefficient().is_one())
150  s->multiply_by_scalar(q->leading_coefficient());
151  if (s->strategy() != nullptr) { s->strategy()->pre_reduction_tasks(tq, *q); }
152  s->add_polynomial_multiple(a, tq, *q, true);
153  }
154  //cout << "after adding other poly: " << *s << endl;
155  return s;
156 }

158 Mutable_Polynomial * Critical_Pair_Dynamic::s_polynomial(
159  SPolyCreationFlags method, StrategyFlags strategy
160 ) {
161  if (s != nullptr) return s;
162  if (p->monomial_ordering() != ordering)
163  p->set_monomial_ordering(ordering);
164  if (q != nullptr and q->monomial_ordering() != ordering)
165  q->set_monomial_ordering(ordering);
166  switch(strategy) {
167  case StrategyFlags::NORMAL_STRATEGY: break; /* nothing to do */
168  case StrategyFlags::SUGAR_STRATEGY: break; /* nothing to do? */
169  case StrategyFlags::WSUGAR_STRATEGY: {
170  if (p->strategy() == nullptr)
171  p->set_strategy(new Poly_WSugar_Data(p, ordering->order_weights()));
172  else {
173  Poly_WSugar_Data * sd = static_cast<Poly_WSugar_Data *>(p->strategy());
174  sd->change_weights(ordering->order_weights());
175  }
176  if (q != nullptr and q->strategy() == nullptr)
177  q->set_strategy(new Poly_WSugar_Data(q, ordering->order_weights()));
178  else if (q != nullptr) {
179  Poly_WSugar_Data * sd = static_cast<Poly_WSugar_Data *>(q->strategy());
180  sd->change_weights(ordering->order_weights());
181  }
182  }
183  default: break; /* default to normal strategy */
184  }
185  return Critical_Pair_Basic::s_polynomial(method, strategy);
186 };
