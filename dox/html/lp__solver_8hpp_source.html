646  for (
647  auto riter = rays.begin();
648  inconsistent and riter != rays.end();
649  ++riter
650  ) {
651  int d = 0;
652  for (int i = 0; i < riter->get_dimension(); ++i)
653  d += (*riter)[i] * (t.degree(i) - u.degree(i));
654  if (d > 0)
655  inconsistent = false;
656  if (show_data) {
657  cout << d << ' ';
658  if (!inconsistent) cout << *riter << endl;
659  }
660  }
661  if (show_data) cout << endl;
662  return not inconsistent;
663  }
