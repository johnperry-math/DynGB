652  bool inconsistent = true;
653  for (
654  auto riter = rays.begin();
655  inconsistent and riter != rays.end();
656  ++riter
657  ) {
658  int d = 0;
659  for (int i = 0; i < riter->get_dimension(); ++i)
660  d += (*riter)[i] * (t.degree(i) - u.degree(i));
661  if (d > 0)
662  inconsistent = false;
663  if (show_data) {
664  cout << d << ' ';
665  if (!inconsistent) cout << *riter << endl;
666  }
667  }
668  if (show_data) cout << endl;
669  return not inconsistent;
670  }
