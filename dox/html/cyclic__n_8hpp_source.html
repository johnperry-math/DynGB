57 list<Abstract_Polynomial *> cyclic_n(
58  NVAR_TYPE n, Prime_Field & F, bool homog,
59  Monomial_Ordering * mord = generic_grevlex_ptr
60 ) {
61  list <Abstract_Polynomial *> result;
62  // set up coefficients and monomials
63  Polynomial_Ring * R = (homog) ? new Polynomial_Ring(n+1, F)
64  : new Polynomial_Ring(n, F);
65  NVAR_TYPE max_n = (homog) ? n + 1 : n;
66  Prime_Field_Element * A = static_cast<Prime_Field_Element *>(
67  malloc(sizeof(Prime_Field_Element) * n)
68  );
69  Monomial * M = static_cast<Monomial *>(calloc(n, sizeof(Monomial)));
70  for (NVAR_TYPE i = 0; i < n; ++i) {
71  M[i].common_initialization();
72  M[i].initialize_exponents(max_n);
73  M[i].set_monomial_ordering(mord);
74  }
75  // ith polynomial for i = 1, ... n-1
76  for (NVAR_TYPE i = 0; i < n - 1; ++i) {
77  // jth monomial...
78  for (NVAR_TYPE j = 0; j < n; ++j)
79  {
80  A[j] = F.unity();
81  // clear exponents first...
82  for (NVAR_TYPE k = 0; k < max_n; ++k)
83  M[j].set_exponent(k,0);
84  // set relevant exponents to 1
85  for (NVAR_TYPE k = j; k < i + j + 1; ++k)
86  {
87  NVAR_TYPE l = (k >= n) ? k - n : k;
88  M[j].set_exponent(l, 1);
89  }
90  }
91  Constant_Polynomial * f = new Constant_Polynomial(n, *R, M, A);
92  f->sort_by_order();
93  result.push_back(f);
94  }
95  // last polynomial has a different structure so we can't run it in the loop
96  for (NVAR_TYPE i = 0; i < max_n; ++i) {
97  if (!homog or i < max_n - 1) {
98  M[0].set_exponent(i, 1);
99  M[1].set_exponent(i, 0);
100  }
101  else {
102  M[0].set_exponent(i, 0);
103  M[1].set_exponent(i, n);
104  }
105  }
106  M[0].set_monomial_ordering(mord);
107  M[0].set_monomial_ordering(mord);
108  A[0] = F.unity();
109  A[1] = -A[0];
110  result.push_back(new Constant_Polynomial(2, *R, M, A));
111  for (DEG_TYPE i = 0; i < n; ++i) M[i].deinitialize();
112  free(M);
113  free(A);
114  return result;
115 }
