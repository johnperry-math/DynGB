43 bool is_zero_base_case(const list<Monomial> & T) {
44  bool result = true; // innocent until proven guilty
45  const Monomial & t = T.front();
46  NVAR_TYPE n = t.num_vars();
47  bool * d = new bool [n] {false};
48  for (const Monomial & t : T) {
49  bool found = false;
50  for (NVAR_TYPE k = 0; result and k < n; ++k) {
51  if (t.degree(k) != 0) {
52  if (found) { result = false; }
53  else {
54  found = true;
55  if (d[k]) { result = false; }
56  else { d[k] = true; }
57  }
58  }
59  }
60  }
61  delete [] d;
62  return result;
63 }

78 Dense_Univariate_Integer_Polynomial * solve_zero_base_case(
79  const list<Monomial> & T,
80  const WT_TYPE * grading
81 ) {
82  DEG_TYPE n = T.front().weighted_degree(grading) + 1;
83  DEG_TYPE d = n;
84  list<Monomial>::const_iterator ti = T.begin();
85  for (++ti; ti != T.end(); ++ti)
86  {
87  n += ti->weighted_degree(grading);
88  if (ti->weighted_degree(grading) + 1 > d)
89  d = ti->weighted_degree(grading) + 1;
90  }
91  Dense_Univariate_Integer_Polynomial * result
92  = new Dense_Univariate_Integer_Polynomial(n);
93  result->set_coefficient(0, 1);
94  Dense_Univariate_Integer_Polynomial * intermediate
95  = new Dense_Univariate_Integer_Polynomial(d);
96  intermediate->set_coefficient(0, 1);
97  for (ti = T.begin(); ti != T.end(); ++ti) {
98  intermediate->set_coefficient(ti->weighted_degree(grading), -1);
99  result->multiply_by(*intermediate);
100  intermediate->set_coefficient(ti->weighted_degree(grading), 0);
101  }
102  delete intermediate;
103  return result;
104 }

123 Dense_Univariate_Integer_Polynomial * solve_one_base_case(
124  const list<Monomial> & T, list<Monomial>::const_iterator ui,
125  const WT_TYPE * grading
126 ) {
127  // copy T and remove the monomial that isn't a simple power
128  list<Monomial> U(T);
129  Monomial p = *ui;
130  U.erase(ui);
131  Dense_Univariate_Integer_Polynomial * result = solve_zero_base_case(U, grading);
132  DEG_TYPE d = 0;
133  DEG_TYPE max_d = 0;
134  for (const Monomial & u : U)
135  {
136  bool found_index = false;
137  unsigned j = 0;
138  for (/* already initialized */ ; not found_index and j < u.num_vars(); ++j)
139  if (u.degree(j) != 0)
140  found_index = true;
141  --j; // to compensate for a forced increment at the end of the loop
142  DEG_TYPE e = (u.degree(j) > p.degree(j)) ?
143  u.degree(j) - p.degree(j) : 0;
144  d += (grading == nullptr) ? e : grading[j] * e;
145  if (max_d < e)
146  max_d = e;
147  }
148  Dense_Univariate_Integer_Polynomial * prod
149  = new Dense_Univariate_Integer_Polynomial(d + p.total_degree() + 1);
150  Dense_Univariate_Integer_Polynomial * fac
151  = new Dense_Univariate_Integer_Polynomial(max_d + 1);
152  fac->set_coefficient(0, 1);
153  prod->set_coefficient(0, 1);
154  for (const Monomial & u : U)
155  {
156  NVAR_TYPE k = 0;
157  for (/* */; k < u.num_vars() and u.degree(k) == 0; ++k) {
158  /* do nothing */
159  }
160  //cout << "one base case: " << *ui << " and " << p << endl;
161  if (grading == nullptr) {
162  fac->set_coefficient(u.degree(k) - p.degree(k), -1);
163  prod->multiply_by(*fac);
164  fac->set_coefficient(u.degree(k) - p.degree(k), 0);
165  } else {
166  WT_TYPE d = grading[k];
167  fac->set_coefficient(d*(u.degree(k) - p.degree(k)), -1);
168  prod->multiply_by(*fac);
169  fac->set_coefficient(d*(u.degree(k) - p.degree(k)), 0);
170  }
171  }
172  delete fac;
173  prod->multiply_by_monomial_of_degree(p.weighted_degree(grading));
174  prod->negate();
175  result->add(*prod);
176  delete(prod);
177  return result;
178 }

180 list<Monomial>::const_iterator is_splitting_case(const list<Monomial> & T) {
181  list<Monomial>::const_iterator result = T.end(); // guilty until proven innocent
182  for (
183  list<Monomial>::const_iterator ti = T.begin();
184  result == T.end() and ti != T.end();
185  ++ti
186  ) {
187  // check whether other monomials are relatively prime to t
188  bool relprime = true;
189  for (
190  list<Monomial>::const_iterator ui = T.begin();
191  relprime and ui != T.end();
192  ++ui
193  ) {
194  if (ti != ui)
195  relprime = ti->is_coprime(*ui);
196  }
197  if (relprime)
198  result = ti;
199  }
200  return result;
201 }

220 Monomial choose_hilbert_pivot(const list<Monomial> & T) {
221  NVAR_TYPE n = T.front().num_vars();
222  unsigned * xcount = new unsigned [n] {0};
223  for (const Monomial & t : T)
224  for (NVAR_TYPE k = 0; k < n; ++k)
225  if (t.degree(k) != 0)
226  ++xcount[k];
227  int i = 0;
228  for (NVAR_TYPE k = 1; k < n; ++k)
229  if (xcount[k] > xcount[i])
230  i = k;
231  delete [] xcount;
232  unsigned * td = new unsigned [T.size()];
233  unsigned j = 0;
234  for (const Monomial & t : T)
235  if (t.degree(i) != 0)
236  td[j++] = t.degree(i);
237  unsigned m = j;
238  std::sort(td, td + m);
239  j = (m == 2) ? 0 : m / 2;
240  Monomial result(n);
241  result.set_exponent(i, td[j]);
242  delete [] td;
243  return result;
244 }

246 Dense_Univariate_Integer_Polynomial * hilbert_numerator_bigatti(
247  const list<Monomial> & T, const WT_TYPE * grading
248 ) {
249  bool verbose = false;
250  if (verbose) {
251  cout << "T = { ";
252  for (const Monomial & t : T) cout << t << " ,";
253  cout << "}\n";
254  if (grading != nullptr) {
255  cout << "w = { ";
256  for (NVAR_TYPE i = 0; i < T.front().num_vars(); ++i)
257  cout << grading[i] << ", ";
258  cout << "}\n";
259  }
260  }
261  Dense_Univariate_Integer_Polynomial * result;
262  list<Monomial>::const_iterator ti;
263  if (T.size() == 1)
264  result = solve_one_monomial_case(T, grading);
265  else if (is_zero_base_case(T))
266  result = solve_zero_base_case(T, grading);
267  else if ((ti = is_one_base_case(T)) != T.end())
268  result = solve_one_base_case(T, ti, grading);
269  else if ((ti = is_splitting_case(T)) != T.end())
270  result = solve_splitting_case(T, ti, grading);
271  else {
272  Monomial p = choose_hilbert_pivot(T);
273  if (verbose) cout << "pivot = " << p << endl;
274  // find pivot precisely
275  unsigned pi = 0;
276  while (p.degree(pi) == 0) { ++pi; }
277  list<Monomial> U, V;
278  for (const Monomial & t : T) {
279  if (not t.divisible_by(p))
280  U.push_back(t);
281  }
282  V = colon_ideal_without_ideals(T, p);
283  U.push_back(p);
284  if (verbose) {
285  cout << "U = { ";
286  for (const Monomial & u : U) cout << u << " ,";
287  cout << "}\n";
288  }
289  if (verbose) {
290  cout << "V = { ";
291  for (const Monomial & v : V) cout << v << " ,";
292  cout << "}\n";
293  }
294  result = hilbert_numerator_bigatti(V, grading);
295  if (verbose) cout << "result from V: " << *result << endl;
296  result->multiply_by_monomial_of_degree(p.weighted_degree(grading));
297  Dense_Univariate_Integer_Polynomial * other
298  = hilbert_numerator_bigatti(U, grading);
299  if (verbose) cout << "result from U: " << *other << endl;
300  *result += *other;
301  delete other;
302  }
303  return result;
304 }

306 Dense_Univariate_Integer_Polynomial * hilbert_second_numerator(
307  NVAR_TYPE n,
308  Dense_Univariate_Integer_Polynomial * first,
309  const WT_TYPE * grading
310 ) {
311  Dense_Univariate_Integer_Polynomial * hn
312  = new Dense_Univariate_Integer_Polynomial(*first);
313  if (grading == nullptr) {
314  unsigned r = 0;
315  NVAR_TYPE i = 0;
316  // perform synthetic division on the Hilbert numerator
317  for (/* */; r == 0 and i < n; ++i) {
318  COEF_TYPE a = hn->coeff(hn->degree());
319  Dense_Univariate_Integer_Polynomial * tmp
320  = new Dense_Univariate_Integer_Polynomial(*hn);
321  for (DEG_TYPE j = hn->degree(); j != 0; --j) {
322  COEF_TYPE b = hn->coeff(j - 1);
323  a += b;
324  hn->set_coefficient(j - 1, a);
325  }
326  if ((r = hn->coeff(0)) != 0) {
327  delete hn;
328  hn = tmp;
329  } else {
330  delete tmp;
331  for (unsigned j = 0; j < hn->degree(); ++j)
332  hn->set_coefficient(j, hn->coeff(j+1));
333  hn->set_coefficient(hn->degree(), 0);
334  }
335  }
336  } else {
337  // perform long division (somewhat optimized) with max unused grading
338  // first find max grading
339  WT_TYPE curr_grad = grading[0];
340  NVAR_TYPE curr_grad_index = 0;
341  for (unsigned k = 1; k < n; ++k)
342  if (grading[k] > curr_grad) {
343  curr_grad = grading[k];
344  curr_grad_index = k;
345  }
346  // divide using max grading
347  for (NVAR_TYPE i = 1; hn->degree() != 0 and i < n; ++i) {
348  Dense_Univariate_Integer_Polynomial * tmp
349  = new Dense_Univariate_Integer_Polynomial(*hn);
350  Dense_Univariate_Integer_Polynomial * q
351  = new Dense_Univariate_Integer_Polynomial(hn->degree());
352  for (DEG_TYPE j = hn->degree(); j > curr_grad - 1; --j) {
353  COEF_TYPE a = hn->coeff(j);
354  q->set_coefficient(j - curr_grad, a);
355  hn->set_coefficient(j, 0);
356  hn->set_coefficient(j - curr_grad, hn->coeff(j - curr_grad) + a);
357  }
358  if (not hn->is_zero()) {
359  delete q; delete hn;
360  hn = tmp;
361  } else {
362  delete tmp; delete hn;
363  hn = q;
364  }
365  // find next larger gradient (or repeat if same occurs)
366  WT_TYPE tmp_grad = grading[0];
367  NVAR_TYPE tmp_grad_index = 0;
368  bool searching = true;
369  for (unsigned k = 0; searching and k < n; ++k) {
370  if (grading[k] == curr_grad and k > curr_grad_index) {
371  curr_grad_index = k;
372  searching = false;
373  }
374  else if (grading[k] > tmp_grad and grading[k] < curr_grad)
375  {
376  tmp_grad = grading[k];
377  tmp_grad_index = k;
378  }
379  }
380  curr_grad = tmp_grad;
381  curr_grad_index = tmp_grad_index;
382  }
383  }
384  return hn;
385 }

418  }
419  delete q;
420  }
421  return p;

424 Dense_Univariate_Rational_Polynomial * hilbert_polynomial(
425  NVAR_TYPE n,
426  unsigned int pole_order,
427  const list<Monomial> T,
428  Dense_Univariate_Integer_Polynomial * hn,
429  Dense_Univariate_Integer_Polynomial * hn2
430 ) {
431  bool own_hn1 = false;
432  bool own_hn2 = false;
433  if (hn == nullptr) {
434  if (T.size() == 0)
435  return nullptr;
436  else {
437  own_hn1 = true;
438  hn = hilbert_numerator_bigatti(T);
439  }
440  }
441  if (hn2 == nullptr) {
442  own_hn2 = true;
443  hn2 = hilbert_second_numerator(n, hn);
444  }
445  if (pole_order == 0)
446  pole_order = n - (hn->degree() - hn2->degree());
447  Dense_Univariate_Rational_Polynomial * hp
448  = new Dense_Univariate_Rational_Polynomial(pole_order + 1);
449  if (pole_order != 0) {
450  DEG_TYPE d = hn2->degree();
451  COEF_TYPE d1 = pole_order - 1;
452  for (DEG_TYPE i = 0; i <= d; ++i) {
453  //q = polynomial_binomial(d1 - d + i, d1);
454  Dense_Univariate_Rational_Polynomial * q = polynomial_binomial(d1 - i, d1);
455  //q->scale_by(hn2->coeff(d - i));
456  q->scale_by(hn2->coeff(i));
457  hp->add(*q);
458  delete q;
459  }
460  }
461  if (pole_order % 2 != n % 2)
462  hp->negate();
463  if (own_hn1) delete hn;
464  if (own_hn2) delete hn2;
465  return hp;
466 }
