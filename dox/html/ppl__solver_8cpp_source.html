42 void PPL_Solver::setup_rays() {
43  rays.clear();
44  if (not lp->is_empty()) {
45  const Generator_System & G = lp->generators();
46  for (const PPL_Generator & g : G) {
47  if (g.is_ray()) {
48  for (NVAR_TYPE i = 0; i < n; ++i)
49  ray_data[i] = g.coefficient(*(X[i])).get_ui();
50  rays.emplace(n, ray_data);
51  }
52  }
53  }
54 }

56 PPL_Solver::PPL_Solver(NVAR_TYPE num_vars) {
57  n = num_vars;
58  ray_data = new RAYENT_TYPE[n];
59  if (instances == 0)
60  Parma_Polyhedra_Library::initialize();
61  ++instances;
62  lp = new NNC_Polyhedron(n);
63  X = (Variable **)malloc(sizeof(Variable *)*n);
64  for (NVAR_TYPE i = 0; i < n; ++i)
65  X[i] = new Variable(i);
66  Constraint_System cs;
67  for (NVAR_TYPE i = 0; i < n; ++i) {
68  cs.insert(*(X[i]) >= 0);
69  }
70  lp->refine_with_constraints(cs);
71  m = n;
72  setup_rays();
73 }

123 bool PPL_Solver::solve(const Constraint & c) {
124  Linear_Expression ineq;
125  for (NVAR_TYPE i = 0; i < n; ++i) {
126  if (c[i] != 0)
127  ineq += Linear_Expression(c[i]*(*X[i]));
128  }
129  PPL_Constraint pc(ineq >= 0);
130  lp->refine_with_constraint(pc);
131  setup_rays();
132  return (rays.size() > 0);
133 }
