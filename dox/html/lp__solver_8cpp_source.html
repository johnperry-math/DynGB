188 void Ray::simplify_ray()
189 {
190  RAYENT_TYPE * w = coords;
191  RAYENT_TYPE gcd = 0;
192  for (NVAR_TYPE i = 0; i < get_dimension(); ++i)
193  {
194  if (gcd == 0)
195  gcd = w[i];
196  else if (w[i] != 0)
197  {
198  RAYENT_TYPE r = (gcd < w[i] ? gcd : w[i]);
199  RAYENT_TYPE s = (gcd < w[i] ? w[i] : gcd);
200  while (r != 0)
201  {
202  RAYENT_TYPE t = s % r;
203  s = r;
204  r = t;
205  }
206  gcd = s;
207  }
208  }
209  if (gcd > 1)
210  for (NVAR_TYPE i = 0; i < get_dimension(); ++i)
211  w[i] /= gcd;
212 }
