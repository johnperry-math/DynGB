56 int hilbert_cmp(
57  const Dense_Univariate_Integer_Polynomial & hn1,
58  const Dense_Univariate_Rational_Polynomial & hp1,
59  const Dense_Univariate_Integer_Polynomial & hn2,
60  const Dense_Univariate_Rational_Polynomial & hp2
61 ) {
62  int result = 0;
63  Dense_Univariate_Rational_Polynomial HPdiff(hp1);
64  HPdiff -= hp2;
65  if (not HPdiff.is_zero()) {
66  if (HPdiff.numerator(HPdiff.degree()) < 0) result = -1;
67  else result = 1;
68  } else { // use Hilbert series
69  DEG_TYPE i = 0;
70  for (/* */; i <= hn1.degree() and i <= hn2.degree() and hn1[i] == hn2[i]; i++)
71  { /* taken care of in loop */ }
72  if (i > hn1.degree()) {
73  if (i > hn2.degree()) // functions are equal
74  result = 0;
75  else
76  result = 1;
77  } else {
78  if (i > hn2.degree()) result = 1;
79  else {
80  if (hn1[i] < hn2[i]) result = -1;
81  else result = 1;
82  }
83  }
84  }
85  return result;

115 void PP_With_Ideal::compute_number_new_pairs() const {
116  int n = t.num_vars();
117  int m = I.size();
118  const list<Monomial> T = I.generators();
119  const vector<Monomial> I(T.begin(), T.end());
120  num_new_pairs = min_deg = 0;
121  bool * keepers = new bool [m];
122  for (int i = 0; i < m; ++i) keepers[i] = true;
123  // first main loop: apply Buchberger's lcm criterion to new pairs
124  for (int i = 0; i < m; ++i)
125  {
126  // if gcd(t,Ii) == 1 then skip i for the time being
127  if (not t.is_coprime(I[i]))
128  {
129  bool has_divisor = false;
130  for (int j=0; (not has_divisor) and j < m; ++j)
131  {
132  if (i != j and keepers[j])
133  {
134  // if some j satisfies lcm(t,Ij) | lcm(t,Ii) then do not count i
135  has_divisor = true;
136  for (int k=1; has_divisor and k <= n; ++k)
137  if // deg(lcm(t,lm(Ii))) < deg(lcm(t,lm(Ij))) ?
138  (((t[k] > I[i][k]) ? t[k] : I[i][k])
139  < ((t[k] > I[j][k]) ? t[k] : I[j][k]))
140  has_divisor = false;
141  }
142  }
143  if (has_divisor) keepers[i] = false;
144  }
145  }
146  // second main loop: apply Buchberger's gcd criterion to new pairs, count survivors
147  for (int i = 0; i < m; ++i)
148  {
149  if (keepers[i] and not t.is_coprime(I[i]))
150  {
151  int new_deg = 0;
152  // determine deg(lcm(t,Si))
153  for (int k=1; k <= n; ++k)
154  new_deg += (t[k] > I[i][k]) ? t[k] : I[i][k];
155  if (min_deg == 0 or min_deg > new_deg)
156  {
157  min_deg = new_deg; num_new_pairs = 1;
158  }
159  else if (min_deg == new_deg)
160  {
161  ++num_new_pairs;
162  //cout << '\t'; pWrite(pHead(strat->S[i]));
163  }
164  }
165  }
166  delete [] keepers;
167  // cout << "we get " << num_new_pairs << " from "; pWrite(t);
168  // third main loop: apply Buchberger's lcm criterion to old pairs, UNLESS
169  // all three lcm's are equal
170  for (Critical_Pair_Dynamic * p : pairs) {
171  const Monomial & u = p->lcm();
172  int new_deg = 0;
173  for (int k=1; k <= n; ++k) new_deg += u[k];
174  // for (int k=1; k <= n; ++k) new_deg += pGetExp(u,k) * Rx->wvhdl[0][k];
175  // no point continuing if it wouldn't change min_deg
176  if (min_deg == 0 or new_deg <= min_deg)
177  {
178  // see if new poly divides this one
179  bool has_divisor = true;
180  // see if Buchberger triple has same lcm
181  bool all_equal = true;
182  const Monomial & p1 = p->first()->leading_monomial();
183  const Monomial & p2 = (p->second() == nullptr) ?
184  p->first()->leading_monomial()
185  : p->second()->leading_monomial();
186  for (int k=1; has_divisor and all_equal and k <= n; ++k)
187  {
188  if (t[k] > u[k]) has_divisor = false;
189  else
190  {
191  // check lcm(t,lm(p1)) == lcm(t,lm(p2)) == lcm(lm(p1),lm(p2)) in xk
192  int a = (t[k] > p1[k]) ? t[k] : p1[k];
193  int b = (t[k] > p2[k]) ? t[k] : p2[k];
194  int c = (p1[k] > p2[k]) ? p1[k] : p2[k];
195  all_equal = (a == c) and (b == c);
196  }
197  }
198  if (not has_divisor or all_equal)
199  {
200  if (min_deg == 0 or min_deg > new_deg)
201  {
202  min_deg = new_deg; num_new_pairs = 1;
203  }
204  else // the only reason we'd be here is if min_deg == new_deg
205  {
206  ++num_new_pairs;
207  //cout << '\t'; pWrite(u);
208  }
209  }
210  }
211  }
212  // cout << " which makes " << num_new_pairs << " pairs total at degree " << min_deg << ".\n";
213 }
