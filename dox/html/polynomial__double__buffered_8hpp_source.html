216  void expand_buffer(unsigned b, unsigned n) {
217  if (mons[b] != nullptr) {
218  for (unsigned k = 0; k < sizes[b]; ++k)
219  mons[b][k].deinitialize();
220  free(mons[b]);
221  free(coeffs[b]);
222  }
223  unsigned new_size = 2*n;
224  sizes[b] = new_size;
225  mons[b] = static_cast<Monomial *>(malloc(new_size*sizeof(Monomial)));
226  for (unsigned k = 0; k < new_size; ++k) {
227  mons[b][k].initialize_exponents(number_of_variables());
228  mons[b][k].clear_ordering_data();
229  }
230  coeffs[b] = static_cast<Prime_Field_Element *>(
231  malloc(new_size*sizeof(Prime_Field_Element))
232  );
233  }
