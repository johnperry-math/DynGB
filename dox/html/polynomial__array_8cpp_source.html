54 Constant_Polynomial::Constant_Polynomial(
55  unsigned length,
56  Polynomial_Ring & R,
57  const Monomial *mons,
58  const Prime_Field_Element *coeffs,
59  const Monomial_Ordering * order
60 ) : Abstract_Polynomial(R, order)
61 {
62  if (order == nullptr) {
63  if (mons[0].monomial_ordering() == nullptr)
64  order = generic_grevlex_ptr;
65  else
66  order = mons[0].monomial_ordering();
67  }
68  m = length;
69  M = static_cast<Monomial *>(calloc(m, sizeof(Monomial)));
70  A = static_cast<Prime_Field_Element *>(calloc(m, sizeof(Prime_Field_Element)));
71  for (unsigned i = 0; i < m; ++i) {
72  M[i].common_initialization();
73  M[i].initialize_exponents(mons[i].num_vars());
74  M[i] = mons[i];
75  if (order != M[i].monomial_ordering())
76  M[i].set_monomial_ordering(order);
77  A[i] = coeffs[i];
78  }
79  head = 0;
80 }

82 Constant_Polynomial::Constant_Polynomial(
83  unsigned length,
84  Polynomial_Ring & R,
85  const vector<Monomial *> mons,
86  const COEF_TYPE *coeffs,
87  unsigned start
88 ) : Abstract_Polynomial(R, mons[0]->monomial_ordering())
89 {
90  m = length;
91  M = static_cast<Monomial *>(calloc(m, sizeof(Monomial)));
92  A = static_cast<Prime_Field_Element *>(calloc(m, sizeof(Prime_Field_Element)));
93  auto F(R.ground_field());
94  auto n = R.number_of_variables();
95  Prime_Field_Element scale(F.inverse(coeffs[start]), &F);
96  unsigned j = 0;
97  for (unsigned i = start; j < m and i < mons.size(); ++i) {
98  if (coeffs[i] != 0) {
99  M[j].common_initialization();
100  M[j].initialize_exponents(n);
101  M[j] = *(mons[i]);
102  A[j] = scale * coeffs[i];
103  ++j;
104  }
105  }
106  head = 0;
107 }

109 Constant_Polynomial::Constant_Polynomial(
110  Polynomial_Ring & R,
111  const list<Monomial> & mons,
112  const list<Prime_Field_Element> & coeffs,
113  const Monomial_Ordering * order
114 ) : Abstract_Polynomial(R, order)
115 {
116  if (order == nullptr) {
117  if (mons.front().monomial_ordering() == nullptr)
118  order = generic_grevlex_ptr;
119  else
120  order = mons.front().monomial_ordering();
121  }
122  m = mons.size();
123  M = static_cast<Monomial *>(calloc(m, sizeof(Monomial)));
124  A = static_cast<Prime_Field_Element *>(calloc(m, sizeof(Prime_Field_Element)));
125  auto Mi = mons.begin();
126  auto Ai = coeffs.begin();
127  unsigned i = 0;
128  while (Mi != mons.end()) {
129  M[i].common_initialization();
130  M[i].initialize_exponents(Mi->num_vars());
131  M[i] = *Mi;
132  if (order != Mi->monomial_ordering())
133  M[i].set_monomial_ordering(order);
134  A[i] = *Ai;
135  ++Mi; ++Ai; ++i;
136  }
137  head = 0;
138 }

157 Constant_Polynomial::Constant_Polynomial(
158  Polynomial_Ring & R,
159  const Monomial_Ordering * order,
160  uint64_t size,
161  uint64_t * AM
162 ) : Abstract_Polynomial(R, order) {
163  m = size;
164  const Prime_Field & F = R.ground_field();
165  M = static_cast<Monomial *>(malloc(m*sizeof(Monomial)));
166  A = static_cast<Prime_Field_Element *>(malloc(m*sizeof(Prime_Field_Element)));
167  unsigned j = 0;
168  for (unsigned i = 0; i < m; ++i) {
169  A[i].assign(AM[j++], &F);
170  M[i].common_initialization();
171  M[i].initialize_exponents(number_of_variables());
172  for (NVAR_TYPE k = 0; k < number_of_variables(); ++k)
173  M[i].set_exponent(k, AM[j++]);
174  M[i].set_monomial_ordering(order);
175  }
176  head = 0;
177 }

179 uint64_t * Constant_Polynomial::serialized(uint64_t & size) {
180  NVAR_TYPE n = number_of_variables();
181  /*uint64_t * result = (uint64_t *)malloc(
182  sizeof(uint64_t)*m*(n+1)
183  );*/
184  uint64_t * result = new uint64_t[m*(n+1)];
185  unsigned j = 0;
186  for (unsigned i = 0; i < m; ++i) {
187  result[j++] = A[i].value();
188  for (NVAR_TYPE k = 0; k < n; ++k)
189  result[j++] = M[i][k];
190  }
191  size = m*(n+1);
192  return result;
193 }

210 void Constant_Polynomial::sort_by_order()
211 {
212  // insertion sort, as we don't expect worst case
213  for (int i = head + 1; i < m; ++i) {
214  int j = i;
215  Monomial t(M[i]);
216  Prime_Field_Element c(A[i]);
217  for (/* */; j > head and t > M[j-1]; --j) {
218  A[j] = A[j-1];
219  M[j] = M[j-1];
220  }
221  if (i != j) {
222  A[j] = c;
223  M[j] = t;
224  }
225  }
226 }

275 Constant_Polynomial::Constant_Polynomial(const Abstract_Polynomial & p)
276 : Abstract_Polynomial(p.base_ring(), p.monomial_ordering())
277 {
278  Polynomial_Iterator * pi = p.new_iterator();
279  m = p.length();
280  M = static_cast<Monomial *>(malloc(m*sizeof(Monomial)));
281  A = static_cast<Prime_Field_Element *>(malloc(m*sizeof(Prime_Field_Element)));
282  for (unsigned i = 0; i < m and !pi->fellOff(); pi->moveRight())
283  {
284  M[i].common_initialization();
285  M[i].initialize_exponents(pi->currMonomial().num_vars());
286  M[i] = pi->currMonomial();
287  A[i] = pi->currCoeff();
288  ++i;
289  }
290  head = 0;
291  delete pi;
292 }
