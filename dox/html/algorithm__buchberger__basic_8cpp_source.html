25 bool lcm_alike(const Monomial & t, const Monomial & u,
26  const Critical_Pair_Basic * p)
27 {
28  bool result = true;
29  for (NVAR_TYPE i = 0; result and i < t.num_vars(); ++i)
30  result = result and p->lcm_degree(i) == ((t.degree(i) >= u.degree(i)) ?
31  t.degree(i) : u.degree(i));
32  return result;
33 }

35 list<Abstract_Polynomial *> reduce_basis(list<Abstract_Polynomial *>G) {
36  list <Abstract_Polynomial *> result;
37  set <Abstract_Polynomial *> eliminate; // save polynomials for deletion
38  // we identify polynomials that need deletion, but wait to delete
39  for (Abstract_Polynomial * f : G)
40  {
41  bool eliminated = false;
42  // look for divisors of f in G
43  for (Abstract_Polynomial * g : G) {
44  if (g != f and g->leading_monomial() | f->leading_monomial()) {
45  eliminated = true;
46  eliminate.insert(f);
47  break;
48  }
49  }
50  // look for divisors of f in result
51  if (not eliminated)
52  for (Abstract_Polynomial * g : result) {
53  if (g != f and g->leading_monomial() | f->leading_monomial()) {
54  eliminated = true;
55  eliminate.insert(f);
56  break;
57  }
58  }
59  if (not eliminated) {
60  result.push_back(f);
61  }
62  }
63  // eliminate the eliminated
64  for (Abstract_Polynomial * g : eliminate)
65  delete g;
66  return result;
67 }

87 void gm_update(
88  list<Critical_Pair_Basic *> & P,
89  list<Abstract_Polynomial *> & G,
90  Abstract_Polynomial * r,
91  StrategyFlags strategy
92 ) {
93  //cout << "----------------------\n";
94  list<Critical_Pair_Basic *> C;
95  // critical pairs with new polynomial
96  for (Abstract_Polynomial * g : G)
97  C.push_back(new Critical_Pair_Basic(g, r, strategy));
98  // apply Buchberger's lcm criterion to new pairs
99  list<Critical_Pair_Basic *> D;
100  while (C.size() != 0) {
101  Critical_Pair_Basic * p = C.front();
102  C.pop_front();
103  if ((p->first()->leading_monomial().is_coprime(
104  p->second()->leading_monomial()))
105  or (no_triplet(p, C) and no_triplet(p, D))
106  )
107  D.push_back(p);
108  else {
109  delete p;
110  //cout << "triplet prunes " << *p << endl;
111  }
112  }
113  // apply Buchberger's gcd criterion
114  list<Critical_Pair_Basic *> E;
115  while (D.size() != 0) {
116  Critical_Pair_Basic * p = D.front();
117  D.pop_front();
118  if (!(p->first()->leading_monomial().is_coprime(
119  p->second()->leading_monomial())))
120  E.push_back(p);
121  else {
122  delete p;
123  // cout << "gcd prunes " << *p << endl;
124  }
125  }
126  // apply Buchberger's lcm criterion to old pairs
127  list<Critical_Pair_Basic *> Q;
128  while (P.size() != 0) {
129  Critical_Pair_Basic * p = P.front();
130  P.pop_front();
131  bool crit1 = !(r->leading_monomial() | p->lcm());
132  bool crit2 = lcm_alike(p->first()->leading_monomial(), r->leading_monomial(), p);
133  bool crit3 = p->second() != nullptr and
134  lcm_alike(p->second()->leading_monomial(), r->leading_monomial(), p);
135  if ( crit1 or crit2 or crit3)
136  Q.push_back(p);
137  else {
138  if (p->s_polynomial() != nullptr) delete p->s_polynomial();
139  delete p;
140  //cout << "triplet prunes " << *p << endl;
141  }
142  }
143  P = Q;
144  // add new pairs to old pairs
145  for (Critical_Pair_Basic * e : E)
146  P.push_back(e);
147  /*cout << "All pairs:\n";
148  for (auto pi = P.begin(); pi != P.end(); ++pi)
149  cout << '\t' << **pi << endl;
150  cout << "----------------------\n";*/
151  // add new poly to basis
152  G.push_back(r);
153 }

182 list<Constant_Polynomial *> buchberger(
183  const list<Abstract_Polynomial *> &F,
184  SPolyCreationFlags method,
185  StrategyFlags strategy,
186  WT_TYPE * strategy_weights
187 ) {
188  unsigned number_of_spolys = 0;
189  list<Abstract_Polynomial *> G;
190  list<Critical_Pair_Basic *> P;
191  // set up basis with generators
192  for (Abstract_Polynomial * fo : F)
193  {
194  Constant_Polynomial * f = new Constant_Polynomial(*fo);
195  switch(strategy) {
196  case StrategyFlags::NORMAL_STRATEGY: break; // don't need polynomial data
197  case StrategyFlags::SUGAR_STRATEGY:
198  f->set_strategy(new Poly_Sugar_Data(f));
199  break;
200  case StrategyFlags::WSUGAR_STRATEGY:
201  f->set_strategy(new Poly_WSugar_Data(f, strategy_weights));
202  break;
203  default: break; // assume normal strategy
204  }
205  if (f->strategy() != nullptr) { f->strategy()->at_generation_tasks(); }
206  P.push_back(new Critical_Pair_Basic(f, strategy));
207  }
208  // main loop
209  bool verbose = false;
210  bool very_verbose = false;
211  while (!P.empty()) {
212  sort_pairs_by_strategy(P);
213  report_critical_pairs(P);
214  Critical_Pair_Basic * p = P.front();
215  report_front_pair(p, strategy);
216  P.pop_front();
217  // make s-poly
218  Mutable_Polynomial * s = p->s_polynomial(method, strategy);
219  ++number_of_spolys;
220  if (p->is_generator())
221  delete p->first();
222  delete p;
223  // cout << "Reducing s-poly "; s->println();
224  if (!s->is_zero())
225  reduce_over_basis<list<Abstract_Polynomial *>>(&s, G);
226  if (s->is_zero()) {
227  cout << "\treduced to zero\n";
228  delete s;
229  } else {
230  Abstract_Polynomial * r;
231  r = new Constant_Polynomial(*s);
232  // move strategy from s to r
233  r->set_strategy(s->strategy());
234  s->set_strategy(nullptr);
235  delete s;
236  cout << "\tadded " << r->leading_monomial() << endl;
237  if (very_verbose) { cout << "\tadded "; r->println(); }
238  gm_update(P, G, r, strategy);
239  }
240  }
241  cout << number_of_spolys << " s-polynomials computed and reduced\n";
242  // cleanup
243  cout << G.size() << " polynomials before interreduction\n";
244  //check_correctness(G, strategy);
245  G = reduce_basis(G);
246  cout << G.size() << " polynomials after interreduction\n";
247  //set<Constant_Polynomial *, smaller_lm> B;
248  list<Constant_Polynomial *> B;
249  unsigned long num_mons = 0;
250  unsigned long max_mons = 0;
251  for (Abstract_Polynomial * g : G) {
252  unsigned long glen = g->length();
253  num_mons += glen;
254  if (glen > max_mons) max_mons = glen;
255  B.push_back(new Constant_Polynomial(*g));
256  }
257  cout << "tot # monomials: " << num_mons;
258  cout << "max # monomials: " << max_mons;
259  cout << "avg # monomials: " << num_mons / B.size() << endl;
260  return B;
261 }
