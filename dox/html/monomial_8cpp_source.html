90 void Monomial::set_exponent(NVAR_TYPE i, DEG_TYPE e) {
91  exponents[i] = e;
92  if (e == 0) {
93  mask &= ~ (1 << i);
94  #if EPACK
95  if (i < NPACK) emask &= ~MASK[i];
96  #endif
97  }
98  else {
99  mask |= (1 << i);
100  #if EPACK
101  if (i < NPACK) emask |= (MASK[i] & (((uint64_t )((PACKSIZE )e)) << (i*NPACK)));
102  #endif
103  }

305 bool Monomial::divisible_by(const Monomial &other) const {
306  bool result = (n == other.n) and (((~ mask) & other.mask) == 0);
307  #if !EPACK
308  for (NVAR_TYPE i = 0; result and i < n; ++i)
309  result = result and exponents[i] >= other.exponents[i];
310  #else
311  for (NVAR_TYPE i = 0; result and i < NPACK; ++i)
312  result = result
313  and ((emask & MASK[i]) >= (other.emask & MASK[i]));
314  for (NVAR_TYPE i = NPACK; result and i < n; ++i)
315  result = result and exponents[i] >= other.exponents[i];
316  #endif
317  return result;
