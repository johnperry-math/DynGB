90 Pair_WSugar_Strategy::Pair_WSugar_Strategy(Critical_Pair_Basic & cpb)
91 {
92  cp = &cpb;
93  if (cpb.first()->strategy() == nullptr) {
94  const CachedWGrevlex_Ordering * mord
95  = dynamic_cast<const CachedWGrevlex_Ordering *>(cpb.first()->monomial_ordering());
96  const WT_TYPE * w = (mord == nullptr) ? nullptr : mord->order_weights();
97  sugar = cpb.first()->weighted_degree(w);
98  }
99  else {
100  Poly_WSugar_Data * wdata
101  = dynamic_cast<Poly_WSugar_Data *>(cpb.first()->strategy());
102  const WT_TYPE * w = (wdata == nullptr) ? nullptr : wdata->weighting();
103  sugar = wdata->poly_sugar() + cpb.first_multiplier().weighted_degree(w);
104  }
105  if (cpb.second() != nullptr) {
106  const Abstract_Polynomial * q = cpb.second();
107  if (false) cout << "second: " << *q << endl;
108  Poly_WSugar_Data * wdata
109  = dynamic_cast<Poly_WSugar_Data *>(q->strategy());
110  const WT_TYPE * w = (wdata == nullptr) ? nullptr : wdata->weighting();
111  DEG_TYPE second_sugar = wdata->poly_sugar()
112  + cpb.second_multiplier().weighted_degree(w);
113  if (second_sugar > sugar) sugar = second_sugar;
114  }

117 bool Pair_WSugar_Strategy::equivalent(const Pair_Strategy_Data & sd) const {
118  const Pair_WSugar_Strategy * nsd = static_cast<const Pair_WSugar_Strategy *>(&sd);
119  bool result = (sugar == nsd->sugar)
120  and (cp->lcm() == nsd->cp->lcm())
121  and (cp->first_multiplier() == nsd->cp->first_multiplier());
122  if (result) {
123  if (cp->second() == nullptr) result = (nsd->cp->second() == nullptr);
124  else {
125  if (nsd->cp->second() != nullptr) result = false;
126  else
127  result = ((cp->second() == nsd->cp->second())
128  and (cp->second_multiplier() == nsd->cp->second_multiplier()));
129  }
130  }
131  return result;
132 }

134 bool Pair_WSugar_Strategy::first_larger(const Pair_Strategy_Data & sd) const {
135  const Pair_WSugar_Strategy * nsd = static_cast<const Pair_WSugar_Strategy *>(&sd);
136  bool result = false;
137  if (sugar > nsd->sugar)
138  result = true;
139  else if (sugar == nsd->sugar and cp->lcm() > nsd->cp->lcm())
140  result = true;
141  else if (sugar == nsd->sugar and cp->lcm().is_like(nsd->cp->lcm())) {
142  if (cp->first_multiplier() > nsd->cp->first_multiplier())
143  result = true;
144  else if (
145  cp->first_multiplier().is_like(nsd->cp->first_multiplier())
146  ) {
147  if (cp->second() == nullptr) result = false;
148  else if (nsd->cp->second() == nullptr)
149  result = true;
150  else
151  result = (cp->second_multiplier() > nsd->cp->second_multiplier());
152  }
153  }
154  return result;
155 }
