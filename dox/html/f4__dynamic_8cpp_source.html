64 F4_Reduction_Data::F4_Reduction_Data(
65  const WGrevlex * curr_ord,
66  const list<Critical_Pair_Dynamic *> & P,
67  const list<Abstract_Polynomial *> & B,
68  Dynamic_Heuristic method
69 ) : G(B), Rx(P.front()->first()->base_ring()), heur(method) {
70  num_cols = 0;
71  // set up the heuristic
72  switch(heur) {
73  case Dynamic_Heuristic::ORD_HILBERT_THEN_DEG:
74  heuristic_judges_smaller = less_by_hilbert_then_degree; break;
75  case Dynamic_Heuristic::ORD_HILBERT_THEN_LEX:
76  heuristic_judges_smaller = less_by_hilbert; break;
77  case Dynamic_Heuristic::DEG_THEN_ORD_HILBERT:
78  heuristic_judges_smaller = less_by_degree_then_hilbert; break;
79  case Dynamic_Heuristic::GRAD_HILB_THEN_DEG:
80  heuristic_judges_smaller = less_by_grad_hilbert_then_degree; break;
81  case Dynamic_Heuristic::SMOOTHEST_DEGREES:
82  heuristic_judges_smaller = less_by_smoothest_degrees; break;
83  case Dynamic_Heuristic::LARGEST_MAX_COMPONENT:
84  heuristic_judges_smaller = less_by_largest_max_component; break;
85  case Dynamic_Heuristic::MIN_CRIT_PAIRS:
86  heuristic_judges_smaller = less_by_num_crit_pairs; break;
87  case Dynamic_Heuristic::BETTI_HILBERT_DEG:
88  heuristic_judges_smaller = less_by_betti; break;
89  case Dynamic_Heuristic::GRAD_BETTI_HILBERT_DEG:
90  heuristic_judges_smaller = less_by_grad_betti; break;
91  default:
92  heuristic_judges_smaller = less_by_hilbert; break;
93  }
94  // set up the matrix
95  A.clear();
96  head.clear();
97  R_build.clear();
98  M_build.clear();
99  strategies.clear();
100  auto mi = M_build.begin();
101  auto ri = R_build.begin();
102  NVAR_TYPE n = Rx.number_of_variables();
103  mord = curr_ord;
104  // sugar data for each row of the matrix
105  for (auto p : P) {
106  Poly_Sugar_Data * new_sugar = new Poly_Sugar_Data(p->first());
107  strategies.push_back(new_sugar);
108  if (strategies.back() != nullptr) {
109  strategies.back()->at_generation_tasks(p->first_multiplier());
110  if (p->second() != nullptr)
111  strategies.back()->pre_reduction_tasks(p->second_multiplier().log(), *(p->second()));
112  }
113  add_monomials(curr_ord, mi, ri, p->first(), p->first_multiplier(), true);
114  if (p->second() != nullptr) {
115  *ri = const_cast<Abstract_Polynomial *>(p->second());
116  /*cout << "for " << **mi << " selected " << p->second()->leading_monomial() << endl;
117  cout << '\t' << p->lcm() << endl;
118  cout << '\t' << p->first()->leading_monomial() << ", " << p->first_multiplier() << endl;
119  cout << '\t' << p->second()->leading_monomial() << ", " << p->second_multiplier() << endl;*/
120  ++ri; ++mi;
121  add_monomials(curr_ord, mi, ri, p->second(), p->second_multiplier());
122  }
123  }
124  // for each monomial, find an appropriate reducer
125  mi = M_build.begin();
126  ri = R_build.begin();
127  while (mi != M_build.end()) {
128  auto g = G.begin();
129  bool found = ((*ri) != nullptr);
130  while (not found and g != G.end()) {
131  if ((**mi).divisible_by((*g)->leading_monomial())) {
132  found = true;
133  cout << "for " << **mi << " selected " << (*g)->leading_monomial() << endl;
134  *ri = *g;
135  Monomial u(**mi);
136  u /= (*g)->leading_monomial();
137  add_monomials(curr_ord, mi, ri, *g, u);
138  g = G.end();
139  }
140  ++g;
141  }
142  ++mi;
143  ++ri;
144  }
145  // populate
146  initialize_many(P);
147  unsigned els = 0;
148  for (auto Ak : A) els += Ak.size();
149  cout << "saved " << (num_rows*num_cols - els)*100 / (num_rows*num_cols) << "%\n";
150 }

152 void F4_Reduction_Data::initialize_some_rows(
153  const list<Critical_Pair_Dynamic *> & P, unsigned row
154 ) {
155  const unsigned num_cols = M_build.size();
156  const Prime_Field & F = P.front()->first()->ground_field();
157  const COEF_TYPE F0 = 0;
158  for (auto cp : P) {
159  auto p = cp->first();
160  const Monomial & t = cp->first_multiplier();
161  auto pi = p->new_iterator();
162  vector<COEF_TYPE> & Arow = A[row];
163  nonzero_entries[row] = 0;
164  unsigned i = 0;
165  while (not M[i]->like_multiple(pi->currMonomial(), t)) ++i;
166  Arow.resize(num_cols - i, F0);
167  Arow[0] = pi->currCoeff().value();
168  offset[row] = i;
169  head[row] = 0;
170  nonzero_entries[row] = 1;
171  pi->moveRight();
172  for (/* */; not pi->fellOff(); ++i) {
173  while (not M[i]->like_multiple(pi->currMonomial(), t)) ++i;
174  Arow[i - offset[row]] = pi->currCoeff().value();
175  pi->moveRight();
176  ++nonzero_entries[row];
177  }
178  delete pi;
179  delete cp;
180  ++row;
181  }

240  const Abstract_Polynomial *g,

256  t1 = M_build.begin();
257  r1 = R_build.begin();
258  auto t2(t1);
259  ++t2;
260  while (t2 != M_build.end()) {
261  ++t1; ++r1;
262  ++t2;
263  }
264  } else if (**t1 != *t) {
265  M_build.insert(t1, t);
266  R_build.insert(r1, nullptr);
267  --t1; --r1;
268  }
269  }
270  auto t2(t1); auto r2(r1);
271  Polynomial_Iterator * pi = g->new_iterator();
272  pi->moveRight();

345 void F4_Reduction_Data::reduce_my_rows(
346  const set<unsigned> & my_rows
347 ) {
348  NVAR_TYPE n = Rx.number_of_variables();
349  const Prime_Field & F = Rx.ground_field();
350  EXP_TYPE * u = new EXP_TYPE[n]; // exponents of multiplier
351  UCOEF_TYPE mod = F.modulus();
352  unsigned new_nonzero_entries;
353  for (unsigned mi = 0; mi < num_cols; ++mi) {
354  // is this monomial reducible?
355  if (R[mi] != nullptr) {
356  for (unsigned k : my_rows) {
357  vector<COEF_TYPE> & Ak = A[k];
358  // do we need to reduce this poly?
359  if (offset[k] <= mi and Ak[mi - offset[k]] != 0) {
360  unsigned i = mi - offset[k];
361  // get reducer for this monomial
362  const Abstract_Polynomial * g = R[mi];
363  Polynomial_Iterator * gi = g->new_iterator();
364  //if (k == 0) cout << "reducing " << *M[mi] << " by " << *g << endl;
365  // determine multiplier
366  for (NVAR_TYPE l = 0; l < n; ++l)
367  u[l] = (*M[mi])[l] - (gi->currMonomial())[l];
368  red_mutex.lock();
369  vector<pair<unsigned, COEF_TYPE> > & r = R_built[mi];
370  if (r.size() == 0) { // need to create reducer
371  unsigned j = mi;
372  // loop through g's terms
373  while (not gi->fellOff()) {
374  const Monomial & t = gi->currMonomial();
375  while (not (M[j]->like_multiple(u, t))) ++j;
376  r.emplace_back(j, gi->currCoeff().value());
377  gi->moveRight();
378  }
379  }
380  delete gi;
381  red_mutex.unlock();
382  // prepare for reduction
383  new_nonzero_entries = nonzero_entries[k];
384  auto sk = strategies[k];
385  sk->pre_reduction_tasks(u, *g);
386  // determine reduction coefficient
387  COEF_TYPE a(mod - ((Ak[i]*F.inverse(r[0].second)) % mod));
388  // loop through g's terms
389  for (auto ri : r) {
390  unsigned j = ri.first;
391  unsigned l = j - offset[k];
392  bool was_zero = Ak[l] == 0;
393  Ak[l] += a*ri.second; Ak[l] %= mod;
394  if (was_zero)
395  ++new_nonzero_entries;
396  else if (Ak[l] == 0)
397  --new_nonzero_entries;
398  // advance
399  }
400  unsigned & hk = head[k];
401  while (hk < Ak.size() and Ak[hk] == 0) ++hk;
402  nonzero_entries[k] = new_nonzero_entries;
403  //if (k == 0) { cout << '\t'; print_row(k); }
404  }
405  }
406  }
407  }
408  delete [] u;
409 }

431 void F4_Reduction_Data::reduce_my_new_rows(
432  unsigned i,
433  unsigned lhead_i,
434  const Prime_Field & F,
435  const set<unsigned> & to_reduce
436 ) {
437  auto mod = F.modulus();
438  const auto & Ai = A[i];
439  for (auto j : to_reduce) {
440  auto ci = head[i];
441  auto & Aj = A[j];
442  auto cj = head[i] + offset[i] - offset[j]; // pos in A[j] of A[i]'s head
443  if (offset[j] + head[j] > offset[i] + head[i]) head[j] = cj;
444  auto a = Aj[lhead_i - offset[j]];
445  unsigned ops = 0;
446  a *= mod - F.inverse(Ai[lhead_i - offset[i]]);
447  while (cj < Aj.size() and ops < nonzero_entries[i]) {
448  if (Ai[ci] != 0) {
449  bool was_zero = (Aj[cj] == 0);
450  Aj[cj] += a*Ai[ci]; Aj[cj] %= mod;
451  if (was_zero)
452  ++nonzero_entries[j];
453  else if (Aj[cj] == 0)
454  --nonzero_entries[j];
455  ++ops;
456  }
457  ++cj; ++ci;
458  }
459  unsigned & hj = head[j];
460  while (hj < Aj.size() and Aj[hj] == 0) ++hj;
461  if (j == 0) { cout << "reduced row " << j << " by new: "; print_row(j); }
462  }

500  UCOEF_TYPE mod = F.modulus();
501  NVAR_TYPE n = M[0]->num_vars();

520  ++k;
521  }
522  }
523  result.push_back(new Constant_Polynomial(
524  nonzero_entries[i],
525  Rx,
526  M_final, A_final,
527  mord
528  ));
529  result.back()->set_strategy(strategies[i]);
530  strategies[i] = nullptr;
531  free(M_final);
532  free(A_final);
533  }
534  }
