63 F4_Reduction_Data::F4_Reduction_Data(
64  const WGrevlex * curr_ord,
65  const list<Critical_Pair_Dynamic *> & P,
66  const list<Abstract_Polynomial *> & B
67 ) : G(B), Rx(P.front()->first()->base_ring()) {
68  num_cols = 0;
69  A.clear();
70  head.clear();
71  R_build.clear();
72  M_build.clear();
73  strategies.clear();
74  auto mi = M_build.begin();
75  auto ri = R_build.begin();
76  NVAR_TYPE n = Rx.number_of_variables();
77  WT_TYPE * wts = new WT_TYPE[n];
78  for (NVAR_TYPE i = 0; i < n; ++i) wts[i] = 1;
79  mord = new WGrevlex(n, wts);
80  for (auto p : P) {
81  Poly_Sugar_Data * new_sugar = new Poly_Sugar_Data(p->first());
82  strategies.push_back(new_sugar);
83  if (strategies.back() != nullptr) {
84  strategies.back()->at_generation_tasks(p->first_multiplier());
85  if (p->second() != nullptr)
86  strategies.back()->pre_reduction_tasks(p->second_multiplier().log(), *(p->second()));
87  }
88  add_monomials(curr_ord, mi, ri, p->first(), p->first_multiplier(), true);
89  if (p->second() != nullptr) {
90  *ri = const_cast<Abstract_Polynomial *>(p->second());
91  ++ri; ++mi;
92  add_monomials(curr_ord, mi, ri, p->second(), p->second_multiplier());
93  }
94  }
95  // for each monomial, find an appropriate reducer
96  mi = M_build.begin();
97  ri = R_build.begin();
98  while (mi != M_build.end()) {
99  auto g = G.begin();
100  bool found = ((*ri) != nullptr);
101  while (not found and g != G.end()) {
102  if ((**mi).divisible_by((*g)->leading_monomial())) {
103  found = true;
104  *ri = *g;
105  Monomial u(**mi);
106  u /= (*g)->leading_monomial();
107  add_monomials(curr_ord, mi, ri, *g, u);
108  g = G.end();
109  }
110  ++g;
111  }
112  ++mi;
113  ++ri;
114  }
115  initialize_many(P);
116  unsigned els = 0;
117  for (auto Ak : A) els += Ak.size();
118  cout << "saved " << (num_rows*num_cols - els)*100 / (num_rows*num_cols) << "%\n";
119 }

121 void F4_Reduction_Data::initialize_some_rows(
122  const list<Critical_Pair_Dynamic *> & P, unsigned row
123 ) {
124  const unsigned num_cols = M_build.size();
125  const Prime_Field & F = P.front()->first()->ground_field();
126  const COEF_TYPE F0 = 0;
127  for (auto cp : P) {
128  auto p = cp->first();
129  const Monomial & t = cp->first_multiplier();
130  auto pi = p->new_iterator();
131  vector<COEF_TYPE> & Arow = A[row];
132  nonzero_entries[row] = 0;
133  unsigned i = 0;
134  while (not M[i]->like_multiple(pi->currMonomial(), t)) ++i;
135  Arow.resize(num_cols - i, F0);
136  Arow[0] = pi->currCoeff().value();
137  offset[row] = i;
138  head[row] = 0;
139  nonzero_entries[row] = 1;
140  pi->moveRight();
141  for (/* */; not pi->fellOff(); ++i) {
142  while (not M[i]->like_multiple(pi->currMonomial(), t)) ++i;
143  Arow[i - offset[row]] = pi->currCoeff().value();
144  pi->moveRight();
145  ++nonzero_entries[row];
146  }
147  delete pi;
148  delete cp;
149  ++row;
150  }
151 }

204 void F4_Reduction_Data::add_monomials(
205  const WGrevlex * curr_ord,
206  list<Monomial *>::iterator & t1,
207  list<Abstract_Polynomial *>::iterator & r1,
208  const Abstract_Polynomial *g,
209  const Monomial & u,
210  bool new_row
211 ) {
212  const_cast<Abstract_Polynomial *>(g)->set_monomial_ordering(curr_ord);
213  NVAR_TYPE n = g->leading_monomial().num_vars();
214  if (new_row) {
215  t1 = M_build.begin();
216  r1 = R_build.begin();
217  Monomial * t = new Monomial(g->leading_monomial());
218  (*t) *= u;
219  find_position(t1, M_build.end(), r1, *t);
220  if (t1 == M_build.end()) {
221  M_build.push_back(t);
222  R_build.push_back(nullptr);
223  t1 = M_build.begin();
224  r1 = R_build.begin();
225  auto t2(t1);
226  ++t2;
227  while (t2 != M_build.end()) {
228  ++t1; ++r1;
229  ++t2;
230  }
231  } else if (**t1 != *t) {
232  M_build.insert(t1, t);
233  R_build.insert(r1, nullptr);
234  --t1; --r1;
235  }
236  }
237  auto t2(t1); auto r2(r1);
238  Polynomial_Iterator * pi = g->new_iterator();
239  pi->moveRight();
240  while (not (pi->fellOff())) {
241  Monomial * t = new Monomial(pi->currMonomial());
242  (*t) *= u;
243  find_position(t2, M_build.end(), r2, *t);
244  if (t2 == M_build.end()) {
245  M_build.push_back(t);
246  R_build.push_back(nullptr);
247  } else if (**t2 != *t) {
248  M_build.insert(t2, t);
249  R_build.insert(r2, nullptr);
250  }
251  pi->moveRight();
252  }
253  delete pi;
254 }

304 void F4_Reduction_Data::reduce_my_rows(
305  const set<unsigned> & my_rows
306 ) {
307  NVAR_TYPE n = Rx.number_of_variables();
308  const Prime_Field & F = Rx.ground_field();
309  EXP_TYPE * u = new EXP_TYPE[n]; // exponents of multiplier
310  UCOEF_TYPE mod = F.modulus();
311  unsigned new_nonzero_entries;
312  for (unsigned mi = 0; mi < num_cols; ++mi) {
313  // is this monomial reducible?
314  if (R[mi] != nullptr) {
315  for (unsigned k : my_rows) {
316  vector<COEF_TYPE> & Ak = A[k];
317  // do we need to reduce this poly?
318  if (offset[k] <= mi and Ak[mi - offset[k]] != 0) {
319  unsigned i = mi - offset[k];
320  // get reducer for this monomial
321  const Abstract_Polynomial * g = R[mi];
322  Polynomial_Iterator * gi = g->new_iterator();
323  // determine multiplier
324  for (NVAR_TYPE l = 0; l < n; ++l)
325  u[l] = (*M[mi])[l] - (gi->currMonomial())[l];
326  red_mutex.lock();
327  vector<pair<unsigned, COEF_TYPE> > & r = R_built[mi];
328  if (r.size() == 0) { // need to create reducer
329  unsigned j = mi;
330  // loop through g's terms
331  while (not gi->fellOff()) {
332  const Monomial & t = gi->currMonomial();
333  while (not (M[j]->like_multiple(u, t))) ++j;
334  r.emplace_back(j, gi->currCoeff().value());
335  gi->moveRight();
336  }
337  delete gi;
338  }
339  red_mutex.unlock();
340  // prepare for reduction
341  new_nonzero_entries = nonzero_entries[k];
342  auto sk = strategies[k];
343  sk->pre_reduction_tasks(u, *g);
344  // determine reduction coefficient
345  COEF_TYPE a(mod - ((Ak[i]*F.inverse(r[0].second)) % mod));
346  // loop through g's terms
347  for (auto ri : r) {
348  unsigned j = ri.first;
349  unsigned l = j - offset[k];
350  bool was_zero = Ak[l] == 0;
351  Ak[l] += a*ri.second; Ak[l] %= mod;
352  if (was_zero)
353  ++new_nonzero_entries;
354  else if (Ak[l] == 0)
355  --new_nonzero_entries;
356  // advance
357  }
358  unsigned & hk = head[k];
359  while (hk < Ak.size() and Ak[hk] == 0) ++hk;
360  nonzero_entries[k] = new_nonzero_entries;
361  }
362  }
363  }
364  }
365  delete [] u;
366 }

388 void F4_Reduction_Data::reduce_my_new_rows(
389  unsigned i,
390  unsigned lhead_i,
391  const Prime_Field & F,
392  const set<unsigned> & to_reduce
393 ) {
394  auto mod = F.modulus();
395  const auto & Ai = A[i];
396  for (auto j : to_reduce) {
397  auto ci = head[i];
398  auto & Aj = A[j];
399  auto c = ci + offset[i] - offset[j]; // pos in A[j] of A[i]'s head
400  auto a = Aj[c];
401  unsigned ops = 0;
402  a *= mod - F.inverse(Ai[lhead_i]);
403  while (c < Aj.size() and ops < nonzero_entries[i]) {
404  if (Ai[ci] != 0) {
405  bool was_zero = (Aj[c] == 0);
406  Aj[c] += a*Ai[ci]; Aj[c] %= mod;
407  if (was_zero)
408  ++nonzero_entries[j];
409  else if (Aj[c] == 0)
410  --nonzero_entries[j];
411  ++ops;
412  }
413  ++c; ++ci;
414  }
415  unsigned & hj = head[j];
416  while (hj < Aj.size() and Aj[hj] == 0) ++hj;
417  }
418 }

454 vector<Constant_Polynomial *> F4_Reduction_Data::finalize() {
455  vector<Constant_Polynomial *> result;
456  const Prime_Field & F = Rx.ground_field();
457  UCOEF_TYPE mod = F.modulus();
458  NVAR_TYPE n = M[0]->num_vars();
459  for (unsigned i = 0; i < num_rows; ++i) {
460  if (nonzero_entries[i] == 0) {
461  delete strategies[i];
462  strategies[i] = nullptr;
463  } else {
464  vector<COEF_TYPE> & Ai = A[i];
465  Monomial * M_final = (Monomial *)malloc(sizeof(Monomial)*nonzero_entries[i]);
466  Prime_Field_Element * A_final
467  = (Prime_Field_Element *)malloc(sizeof(Prime_Field_Element)*nonzero_entries[i]);
468  COEF_TYPE scale = F.inverse(Ai[head[i]]);
469  unsigned k = 0;
470  for (unsigned j = head[i]; k < nonzero_entries[i]; ++j) {
471  if (Ai[j] != 0) {
472  A_final[k].assign(Ai[j]*scale % mod, &F);
473  M_final[k].common_initialization();
474  M_final[k].initialize_exponents(n);
475  M_final[k].set_monomial_ordering(mord);
476  M_final[k] = *M[offset[i] + j];
477  ++k;
478  }
479  }
480  result.push_back(new Constant_Polynomial(
481  nonzero_entries[i],
482  Rx,
483  M_final, A_final,
484  mord
485  ));
486  result.back()->set_strategy(strategies[i]);
487  strategies[i] = nullptr;
488  free(M_final);
489  free(A_final);
490  }
491  }
492  return result;
493 }

506 WGrevlex * F4_Reduction_Data::select_dynamic(
507  list<Monomial> & U,
508  const list<Abstract_Polynomial *> G,
509  const list<Critical_Pair_Dynamic *> & P,
510  WGrevlex * curr_ord,
511  LP_Solver * & skel,
512  Dynamic_Heuristic heur
513 ) {
514  bool ordering_changed = false;
515  WGrevlex * result = curr_ord;
516  PP_With_Ideal * newideal = nullptr;
517  LP_Solver * winning_skel = nullptr;
518  set<unsigned> unprocessed;
519  vector<set<Monomial> > potential_pps(number_of_rows());
520  for (unsigned i = 0; i < number_of_rows(); ++i) {
521  if (nonzero_entries[i] != 0) {
522  unprocessed.insert(i);
523  monomials_in_row(i, potential_pps[i]);
524  }
525  }
526  Dense_Univariate_Integer_Polynomial *hn = nullptr;
527  // select most advantageous unprocessed poly, reduce others
528  while (unprocessed.size() != 0) {
529  unsigned winning_row = *(unprocessed.begin());
530  for (unsigned i : unprocessed) {
531  unsigned j; // will store location of new leading monomial
532  // copy skeleton
533  LP_Solver * new_lp;
534  list<Monomial> T {U};
535  if (dynamic_cast<Skeleton *>(skel) != nullptr)
536  new_lp = new Skeleton(*dynamic_cast<Skeleton *>(skel));
537  else if (dynamic_cast<GLPK_Solver *>(skel) != nullptr)
538  new_lp = new GLPK_Solver(*dynamic_cast<GLPK_Solver *>(skel));
539  else if (dynamic_cast<PPL_Solver *>(skel) != nullptr)
540  new_lp = new PPL_Solver(*dynamic_cast<PPL_Solver *>(skel));
541  // see if we can obtain a better ordering from this polynomial
542  vector<Dense_Univariate_Integer_Polynomial*> H(number_of_rows());
543  for (auto h : H)
544  h = hn == nullptr ?
545  nullptr : new Dense_Univariate_Integer_Polynomial(*hn);
546  bool new_ordering = false;
547  select_monomial(
548  potential_pps[i], *M[offset[i] + head[i]], U, &H[i], G, P, new_lp,
549  new_ordering, heur
550  );
551  // if we have a new ordering, compare the result to current preference
552  // we use PP_With_Ideal to hold relevant dynamic information
553  if (!new_ordering) {
554  delete new_lp;
555  } else {
556  // first find the index of the leading monomial (need for reduction)
557  Monomial t = U.back();
558  U.pop_back();
559  bool found_mon = false;
560  for (unsigned k = offset[i]; not found_mon; ++k) {
561  if (*M[k] == t) {
562  j = k;
563  found_mon = true;
564  }
565  }
566  // if tempideal beats newideal, reassign newideal
567  Ray r(mord->number_of_weights(), mord->order_weights());
568  if (newideal == nullptr) { // always true in this case
569  ordering_changed = true;
570  winning_row = i;
571  winning_skel = new_lp;
572  delete skel;
573  skel = winning_skel;
574  newideal = new PP_With_Ideal(t, T, r, P, nullptr);
575  newideal->set_hilbert_numerator(H[i]);
576  } else {
577  PP_With_Ideal * tempideal = new PP_With_Ideal(t, T, r, P, nullptr);
578  tempideal->set_hilbert_numerator(H[i]);
579  bool change_winner;
580  switch(heur) {
581  case Dynamic_Heuristic::ORD_HILBERT_THEN_DEG:
582  change_winner = less_by_hilbert_then_degree(*tempideal, *newideal); break;
583  case Dynamic_Heuristic::ORD_HILBERT_THEN_LEX:
584  change_winner = less_by_hilbert(*tempideal, *newideal); break;
585  case Dynamic_Heuristic::DEG_THEN_ORD_HILBERT:
586  change_winner = less_by_degree_then_hilbert(*tempideal, *newideal); break;
587  case Dynamic_Heuristic::GRAD_HILB_THEN_DEG:
588  change_winner = less_by_grad_hilbert_then_degree(*tempideal, *newideal); break;
589  case Dynamic_Heuristic::SMOOTHEST_DEGREES:
590  change_winner = less_by_smoothest_degrees(*tempideal, *newideal); break;
591  case Dynamic_Heuristic::LARGEST_MAX_COMPONENT:
592  change_winner = less_by_largest_max_component(*tempideal, *newideal); break;
593  case Dynamic_Heuristic::MIN_CRIT_PAIRS:
594  change_winner = less_by_num_crit_pairs(*tempideal, *newideal); break;
595  case Dynamic_Heuristic::BETTI_HILBERT_DEG:
596  change_winner = less_by_betti(*tempideal, *newideal); break;
597  case Dynamic_Heuristic::GRAD_BETTI_HILBERT_DEG:
598  change_winner = less_by_grad_betti(*tempideal, *newideal); break;
599  default:
600  change_winner = less_by_hilbert(*tempideal, *newideal); break;
601  }
602  if (change_winner) { // winner has changed; reduce matrix by it
603  ordering_changed = true;
604  delete newideal;
605  winning_row = i;
606  newideal = tempideal;
607  winning_skel = new_lp;
608  delete skel;
609  skel = winning_skel;
610  reduce_by_new(i, j);
611  } // change_winner?
612  } // newideal == nullptr?
613  } // ordering changed?
614  } // loop through rows
615  unprocessed.erase(winning_row);
616  }
617  if (ordering_changed) {
618  skel = winning_skel;
619  result = new WGrevlex(ray_sum(skel->get_rays()));
620  }
621  delete newideal;
622  return result;
623 }

733  g->set_monomial_ordering(curr_ord);

736  }

747  cout << "parallel f4 took " << duration << " seconds\n";
