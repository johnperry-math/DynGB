70 class Grading_Order_Data_Allocator {
71 public:
76  goda_block<TYPE> * allocate_new_block() {
77  if (elaborate) {
78  cout << name << " allocating " << BLOCK_COUNT << " blocks\n";
79  cout << num_blocks + BLOCK_COUNT << " blocks total\n";
80  }
81  goda_block<TYPE> * result
82  = (goda_block<TYPE> *)malloc(BLOCK_COUNT * data_size*sizeof(size_t));
83  goda_block<TYPE> * tmp = result;
84  for (unsigned i=0; i < BLOCK_COUNT - 1; ++i) {
85  tmp[i*data_size].next = &(tmp[(i+1)*data_size]);
86  }
87  tmp[(BLOCK_COUNT - 1)*data_size].next = nullptr;
88  return result;
89  }
98  Grading_Order_Data_Allocator(
99  NVAR_TYPE n, const string & block_name = "", bool verbose=false
100  )
101  : data_size(n), name(block_name), elaborate(verbose)
102  {
103  big_blocks = block = allocate_new_block();
104  block->next = nullptr; block += data_size;
105  }
107  ~Grading_Order_Data_Allocator() {
108  cout << "at termination, " << name << " has " << num_blocks << " blocks\n";
109  while (big_blocks != nullptr) {
110  goda_block<TYPE> * next_block = big_blocks->next;
111  cout << name << " freeing " << BLOCK_COUNT << " blocks\n";
112  if (num_blocks < BLOCK_COUNT) num_blocks = 0;
113  else num_blocks -= BLOCK_COUNT;
114  free(big_blocks);
115  big_blocks = next_block;
116  }
117  cout << name << " destructs with " << num_blocks << " blocks\n";
118  }
123  TYPE * get_new_block() {
124  if (block == nullptr) {
125  block = allocate_new_block();
126  block->next = big_blocks; big_blocks = block; block += data_size;
127  }
128  ++num_blocks;
129  if (elaborate)
130  cout << name << " increase to " << num_blocks << " blocks in use\n";
131  TYPE * result = reinterpret_cast<TYPE *>(block);
132  block = block->next;
133  return result;
134  }
139  void return_used_block(TYPE * freed_block) {
140  goda_block<TYPE> * new_head = (goda_block<TYPE> *)freed_block;
141  --num_blocks;
142  if (elaborate)
143  cout << name << " decrease to " << num_blocks << " blocks in use\n";
144  new_head->next = block;
145  block = new_head;
146  }
150  void report() {

152  }
