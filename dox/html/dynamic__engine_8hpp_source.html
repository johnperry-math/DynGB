219  inline int get_difference_in_degree()
220  {
221  if (min_deg < 0)
222  {
223  const list<Monomial> & G = I.generators();
224  min_deg = max_deg = G.front().weighted_degree(ordering.weights());
225  for (const Monomial & g : G) {
226  int d = g.weighted_degree(ordering.weights());
227  if (d < min_deg) min_deg = d;
228  if (d > max_deg) max_deg = d;
229  }
230  }
231  return max_deg - min_deg;
232  }
