101  if (result) {
102  if (cp->second() == nullptr) result = (nsd->cp->second() == nullptr);
103  else {
104  if (nsd->cp->second() != nullptr) { result = false; }
105  else
106  result = ((cp->second() == nsd->cp->second())

108  }
109  }

113 bool Pair_Sugar_Data::first_larger(const Pair_Strategy_Data & sd) const {
114  const Pair_Sugar_Data * nsd = static_cast<const Pair_Sugar_Data *>(&sd);
115  bool result = false;
116  if (sugar > nsd->sugar)
117  result = true;
118  else if (sugar == nsd->sugar and cp->lcm() > nsd->cp->lcm())
119  result = true;
120  else if (sugar == nsd->sugar and cp->lcm().is_like(nsd->cp->lcm())) {
121  if (cp->first_multiplier() > nsd->cp->first_multiplier())
122  result = true;
123  else if (
124  cp->first_multiplier().is_like(nsd->cp->first_multiplier())
125  ) {
126  if (cp->second() == nullptr) result = false;
127  else if (nsd->cp->second() == nullptr) result = true;
128  else result = true;
129  }
130  }
131  return result;
