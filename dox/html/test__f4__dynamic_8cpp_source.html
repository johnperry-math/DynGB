95 bool meaningful_arguments(
96  int argc, char *argv[],
97  bool & homogeneous, int & modulus, int & numvars
98 ) {
99  modulus = 43;
100  homogeneous = false;
101  WT_TYPE * weights = nullptr;
102  unsigned int order_flag = 0;
103  bool good_args = (argc > 1);
104  if (good_args) {
105  for (int i = 1; good_args and i < argc; ++i) {
106  if (!strcmp(argv[i],"hom") or !strcmp(argv[i],"homog")
107  or !strcmp(argv[i],"homogeneous"))
108  homogeneous = true;
109  else {
110  int j = 0;
111  for (/* */; argv[i][j] != '=' and argv[i][j] != '\0'; ++j) { /* */ }
112  if (argv[i][j] != '=') {
113  good_args = false;
114  cout << "Options must have form <option>=<value>.\n";
115  }
116  else {
117  argv[i][j] = '\0';
118  if (!strcmp(argv[i],"n") or !strcmp(argv[i],"num")
119  or !strcmp(argv[i],"numvars")) {
120  numvars = atoi(&(argv[i][j+1]));
121  if (numvars < 3) {
122  good_args = false;
123  cout << "Invalid number of variables: must be at least 3.\n";
124  }
125  }
126  else if (!strcmp(argv[i],"m") or !strcmp(argv[i],"mod")
127  or !strcmp(argv[i],"modulus"))
128  {
129  modulus = atoi(&(argv[i][j+1]));
130  if (modulus < 2) {
131  good_args = false;
132  cout << "Invalid modulus; must be at least 2.\n";
133  }
134  }
135  else {
136  cout << "Unrecognized argument.\n"; good_args = false;
137  }
138  }
139  }
140  }
141  }
142  return good_args;
143 }
