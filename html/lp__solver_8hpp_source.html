585  ++riter

587  int d = 0;
588  for (int i = 0; i < riter->get_dimension(); ++i)
589  d += (*riter)[i] * (t.degree(i) - u.degree(i));
590  if (d > 0)
591  inconsistent = false;
592  if (show_data) {
593  cout << d << ' ';
594  if (!inconsistent) cout << *riter << endl;
595  }
596  }
